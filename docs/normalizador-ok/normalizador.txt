```javascript
// normalizador.js - Normalizador General de Descargas
// Este archivo es completamente independiente de index.js

class ProjectNormalizer {
    constructor() {
        this.originalHTML = '';
        this.normalizedHTML = '';
        this.combinedCSS = '';
        this.combinedJS = '';
        this.stats = {
            originalLines: 0,
            normalizedLines: 0,
            cssLines: 0,
            jsLines: 0,
            externalCSS: 0,
            externalJS: 0
        };
    }

    /**
     * Normaliza un proyecto descargado para uso comercial
     * @param {string} html - Contenido HTML original
     * @param {string} css - CSS adicional (opcional)
     * @param {string} js - JavaScript adicional (opcional)
     * @returns {Object} - Objeto con HTML normalizado, CSS y JS extraídos
     */
    normalizeDownloadedProject(html, css = '', js = '') {
        this.originalHTML = html;
        this.combinedCSS = css || '';
        this.combinedJS = js || '';
        
        // Calcular estadísticas iniciales
        this.stats.originalLines = html.split('\n').length;
        
        // Proceso de normalización
        this.extractAllCSS();
        this.extractAllJS();
        
        // Reconstruir HTML normalizado
        const result = this.reconstructHTML();
        
        // Calcular estadísticas finales
        this.stats.normalizedLines = result.html.split('\n').length;
        this.stats.cssLines = this.combinedCSS.split('\n').length;
        this.stats.jsLines = this.combinedJS.split('\n').length;
        
        return {
            html: result.html,
            css: this.combinedCSS,
            js: this.combinedJS,
            stats: this.stats
        };
    }

    /**
     * Extrae todo el CSS del HTML
     */
    extractAllCSS() {
        const html = this.originalHTML;
        
        // Extraer CSS interno
        const internalCSS = this.extractInternalCSS(html);
        if (internalCSS) {
            this.combinedCSS += '\n' + internalCSS;
        }
        
        // Extraer y marcar CSS externo
        this.extractExternalCSS(html);
    }

    /**
     * Extrae CSS interno (etiquetas <style>)
     */
    extractInternalCSS(html) {
        const styleRegex = /<style[^>]*>([\s\S]*?)<\/style>/gi;
        const matches = [];
        let match;
        
        while ((match = styleRegex.exec(html)) !== null) {
            matches.push(match[1].trim());
        }
        
        return matches.join('\n');
    }

    /**
     * Identifica y marca CSS externo
     */
    extractExternalCSS(html) {
        const linkRegex = /<link[^>]*href=["']([^"']*\.css)["'][^>]*>/gi;
        let match;
        let externalCount = 0;
        
        while ((match = linkRegex.exec(html)) !== null) {
            const cssPath = match[1];
            this.combinedCSS += `\n/* ========================================== */\n`;
            this.combinedCSS += `/* CSS EXTERNO: ${cssPath} */\n`;
            this.combinedCSS += `/* Este archivo necesita ser incorporado manualmente */\n`;
            this.combinedCSS += `/* Descarga el archivo y pega su contenido aquí */\n`;
            this.combinedCSS += `/* ========================================== */\n\n`;
            externalCount++;
        }
        
        this.stats.externalCSS = externalCount;
    }

    /**
     * Extrae todo el JavaScript del HTML
     */
    extractAllJS() {
        const html = this.originalHTML;
        
        // Extraer JS interno
        const internalJS = this.extractInternalJS(html);
        if (internalJS) {
            this.combinedJS += '\n' + internalJS;
        }
        
        // Extraer y marcar JS externo
        this.extractExternalJS(html);
    }

    /**
     * Extrae JavaScript interno (etiquetas <script> sin src)
     */
    extractInternalJS(html) {
        const scriptRegex = /<script(?!.*src)[^>]*>([\s\S]*?)<\/script>/gi;
        const matches = [];
        let match;
        
        while ((match = scriptRegex.exec(html)) !== null) {
            matches.push(match[1].trim());
        }
        
        return matches.join('\n');
    }

    /**
     * Identifica y marca JavaScript externo
     */
    extractExternalJS(html) {
        const scriptRegex = /<script[^>]*src=["']([^"']*\.js)["'][^>]*><\/script>/gi;
        let match;
        let externalCount = 0;
        
        while ((match = scriptRegex.exec(html)) !== null) {
            const jsPath = match[1];
            this.combinedJS += `\n// ==========================================\n`;
            this.combinedJS += `// JAVASCRIPT EXTERNO: ${jsPath}\n`;
            this.combinedJS += `// Este archivo necesita ser incorporado manualmente\n`;
            this.combinedJS += `// Descarga el archivo y pega su contenido aquí\n`;
            this.combinedJS += `// ==========================================\n\n`;
            externalCount++;
        }
        
        this.stats.externalJS = externalCount;
    }

    /**
     * Reconstruye el HTML con estructura normalizada
     */
    reconstructHTML() {
        let html = this.originalHTML;
        
        // 1. Extraer metadatos importantes
        const title = this.extractTitle(html);
        const metaTags = this.extractMetaTags(html);
        const bodyContent = this.extractBodyContent(html);
        
        // 2. Crear nuevo HTML estructurado
        let normalizedHTML = `<!DOCTYPE html>
<html lang="es">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    ${metaTags}
    <title>${title}</title>`;
    
        // 3. Agregar CSS si existe
        if (this.combinedCSS.trim()) {
            normalizedHTML += '\n    <style>\n' + this.combinedCSS.trim() + '\n    </style>';
        }
        
        normalizedHTML += '\n</head>\n<body>\n';
        
        // 4. Agregar contenido del body
        normalizedHTML += bodyContent;
        
        // 5. Agregar JS si existe (al final del body)
        if (this.combinedJS.trim()) {
            normalizedHTML += '\n    <script>\n' + this.combinedJS.trim() + '\n    </script>';
        }
        
        normalizedHTML += '\n</body>\n</html>';
        
        return {
            html: normalizedHTML,
            css: this.combinedCSS,
            js: this.combinedJS
        };
    }

    /**
     * Extrae el título del documento
     */
    extractTitle(html) {
        const titleMatch = html.match(/<title[^>]*>([\s\S]*?)<\/title>/i);
        return titleMatch ? titleMatch[1].trim() : 'Documento Normalizado';
    }

    /**
     * Extrae metatags importantes
     */
    extractMetaTags(html) {
        const metaTags = [];
        const metaRegex = /<meta[^>]+>/gi;
        let match;
        
        while ((match = metaRegex.exec(html)) !== null) {
            const metaTag = match[0];
            // Incluir solo metatags importantes
            if (metaTag.includes('charset') || 
                metaTag.includes('viewport') || 
                metaTag.includes('description') ||
                metaTag.includes('keywords') ||
                metaTag.includes('author')) {
                metaTags.push(metaTag);
            }
        }
        
        return metaTags.join('\n    ');
    }

    /**
     * Extrae el contenido del body
     */
    extractBodyContent(html) {
        const bodyMatch = html.match(/<body[^>]*>([\s\S]*?)<\/body>/i);
        return bodyMatch ? bodyMatch[1].trim() : html;
    }

    /**
     * Genera un reporte de normalización
     */
    generateReport() {
        return {
            originalLines: this.stats.originalLines,
            normalizedLines: this.stats.normalizedLines,
            cssLines: this.stats.cssLines,
            jsLines: this.stats.jsLines,
            externalCSS: this.stats.externalCSS,
            externalJS: this.stats.externalJS,
            compressionRatio: ((this.stats.originalLines - this.stats.normalizedLines) / this.stats.originalLines * 100).toFixed(2)
        };
    }

    /**
     * Exporta el proyecto normalizado como archivos separados
     */
    exportAsFiles() {
        const timestamp = new Date().toISOString().slice(0, 19).replace(/:/g, '-');
        
        return {
            html: {
                filename: `proyecto-normalizado-${timestamp}.html`,
                content: this.normalizedHTML
            },
            css: {
                filename: `styles-normalized-${timestamp}.css`,
                content: this.combinedCSS
            },
            js: {
                filename: `scripts-normalized-${timestamp}.js`,
                content: this.combinedJS
            },
            report: {
                filename: `reporte-normalizacion-${timestamp}.txt`,
                content: this.generateReportText()
            }
        };
    }

    /**
     * Genera texto del reporte
     */
    generateReportText() {
        const report = this.generateReport();
        return `REPORTE DE NORMALIZACIÓN
========================
Fecha: ${new Date().toLocaleString()}
Archivo original: ${report.originalLines} líneas
Archivo normalizado: ${report.normalizedLines} líneas
CSS extraído: ${report.cssLines} líneas
JavaScript extraído: ${report.jsLines} líneas
Archivos CSS externos identificados: ${report.externalCSS}
Archivos JS externos identificados: ${report.externalJS}
Ratio de compresión: ${report.compressionRatio}%

INSTRUCCIONES:
=============
1. El HTML ahora contiene todo el CSS y JS internos
2. Para mejor rendimiento, guarda CSS y JS en archivos separados
3. Los archivos externos identificados necesitan ser incorporados manualmente
4. Optimiza el código para producción antes de publicar`;
    }
}

// ===== FUNCIONES GLOBALES PARA USO EXTERNO =====

/**
 * Función global para normalizar proyectos
 * @param {string} htmlContent - Contenido HTML a normalizar
 * @returns {Object} - Proyecto normalizado
 */
window.normalizarProyecto = function(htmlContent) {
    const normalizador = new ProjectNormalizer();
    return normalizador.normalizeDownloadedProject(htmlContent);
};

/**
 * Función global para exportar proyecto normalizado
 * @param {string} htmlContent - Contenido HTML a normalizar
 * @returns {Object} - Archivos listos para descargar
 */
window.exportarProyectoNormalizado = function(htmlContent) {
    const normalizador = new ProjectNormalizer();
    const resultado = normalizador.normalizeDownloadedProject(htmlContent);
    return normalizador.exportAsFiles();
};

/**
 * Función global para obtener reporte de normalización
 * @param {string} htmlContent - Contenido HTML a analizar
 * @returns {Object} - Reporte de análisis
 */
window.analizarProyecto = function(htmlContent) {
    const normalizador = new ProjectNormalizer();
    normalizador.normalizeDownloadedProject(htmlContent);
    return normalizador.generateReport();
};

// ===== DETECCIÓN AUTOMÁTICA Y USO CON ACE EDITOR =====

/**
 * Inicializa el normalizador con Ace Editor si está disponible
 */
window.inicializarNormalizador = function() {
    // Verificar si Ace Editor está disponible
    if (typeof ace !== 'undefined' && ace.edit) {
        console.log('✅ Normalizador inicializado con Ace Editor');
        
        // Exponer funciones para el editor
        window.normalizador = {
            normalizar: function() {
                const editor = ace.edit("editor");
                const contenido = editor.getValue();
                return window.normalizarProyecto(contenido);
            },
            
            exportar: function() {
                const editor = ace.edit("editor");
                const contenido = editor.getValue();
                return window.exportarProyectoNormalizado(contenido);
            },
            
            analizar: function() {
                const editor = ace.edit("editor");
                const contenido = editor.getValue();
                return window.analizarProyecto(contenido);
            }
        };
    } else {
        console.log('⚠️ Ace Editor no detectado - Normalizador funciona de forma independiente');
    }
};

// Auto-inicialización cuando el DOM está listo
if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', window.inicializarNormalizador);
} else {
    window.inicializarNormalizador();
}

// Exportación para módulos (si se usa como módulo)
if (typeof module !== 'undefined' && module.exports) {
    module.exports = ProjectNormalizer;
}
```

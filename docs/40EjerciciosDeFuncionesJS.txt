### **Soluciones**

```javascript
// 1. Función que devuelve el mayor de dos números
function encontrarMayor(a, b) {
  return a > b ? a : b;
}

// 2. Función para verificar si un número es par
function esPar(num) {
  return num % 2 === 0;
}

// 3. Función que devuelve el área de un círculo
function areaCirculo(radio) {
  return Math.PI * radio ** 2;
}

// 4. Función para convertir grados Celsius a Fahrenheit
function celsiusAFahrenheit(celsius) {
  return (celsius * 9/5) + 32;
}

// 5. Función que invierte una cadena
function invertirCadena(str) {
  return str.split('').reverse().join('');
}

// 6. Función para contar vocales en una cadena
function contarVocales(str) {
  const vocales = str.match(/[aeiouáéíóú]/gi);
  return vocales ? vocales.length : 0;
}

// 7. Función que encuentra la palabra más larga en un array
function palabraMasLarga(arr) {
  return arr.reduce((larga, actual) => 
    actual.length > larga.length ? actual : larga, "");
}

// 8. Función para generar un número aleatorio en un rango
function numeroAleatorio(min, max) {
  return Math.floor(Math.random() * (max - min + 1)) + min;
}

// 9. Función que calcula el factorial de un número
function factorial(n) {
  if (n === 0) return 1;
  return n * factorial(n - 1);
}

// 10. Función para verificar si una palabra es un palíndromo
function esPalindromo(str) {
  const limpia = str.replace(/[\W_]/g, '').toLowerCase();
  return limpia === invertirCadena(limpia);
}

// 11. Función que suma todos los números de un array
function sumarArray(arr) {
  return arr.reduce((suma, num) => suma + num, 0);
}

// 12. Función para encontrar la diferencia de edad entre dos personas
function diferenciaEdad(edad1, edad2) {
  return Math.abs(edad1 - edad2);
}

// 13. Función que convierte horas a segundos
function horasASegundos(horas) {
  return horas * 3600;
}

// 14. Función para calcular el IMC
function calcularIMC(peso, altura) {
  return peso / (altura ** 2);
}

// 15. Función que capitaliza la primera letra de una cadena
function capitalizar(str) {
  return str.charAt(0).toUpperCase() + str.slice(1);
}

// 16. Función para verificar si un número es primo
function esPrimo(num) {
  if (num <= 1) return false;
  for (let i = 2; i <= Math.sqrt(num); i++) {
    if (num % i === 0) return false;
  }
  return true;
}

// 17. Función que elimina los elementos falsy de un array
function eliminarFalsy(arr) {
  return arr.filter(Boolean);
}

// 18. Función para ordenar un array de números de forma ascendente
function ordenarAscendente(arr) {
  return arr.slice().sort((a, b) => a - b);
}

// 19. Función que cuenta la cantidad de letras en una cadena
function contarLetras(str) {
  return str.replace(/[^a-záéíóúñ]/gi, '').length;
}

// 20. Función para generar una contraseña aleatoria
function generarPassword() {
  const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
  let password = '';
  for (let i = 0; i < 8; i++) {
    password += chars.charAt(Math.floor(Math.random() * chars.length));
  }
  return password;
}

// 21. Función que calcula el promedio de un array de números
function calcularPromedio(arr) {
  return sumarArray(arr) / arr.length;
}

// 22. Función para encontrar la intersección de dos arrays
function interseccionArrays(arr1, arr2) {
  return arr1.filter(item => arr2.includes(item));
}

// 23. Función que repite una cadena n veces
function repetirCadena(str, n) {
  return str.repeat(n);
}

// 24. Función para calcular la potencia de un número
function potencia(base, exp) {
  return base ** exp;
}

// 25. Función que devuelve el tipo de dato de un valor
function tipoDeDato(valor) {
  return typeof valor;
}

// 26. Función para eliminar duplicados de un array
function eliminarDuplicados(arr) {
  return [...new Set(arr)];
}

// 27. Función que convierte una cadena a camelCase
function aCamelCase(str) {
  return str.replace(/[-_](.)/g, (_, c) => c.toUpperCase());
}

// 28. Función para obtener la fecha actual en formato dd/mm/aaaa
function fechaActual() {
  const fecha = new Date();
  return `${fecha.getDate()}/${fecha.getMonth() + 1}/${fecha.getFullYear()}`;
}

// 29. Función que calcula la hipotenusa de un triángulo rectángulo
function calcularHipotenusa(a, b) {
  return Math.sqrt(a * a + b * b);
}

// 30. Función para verificar si un año es bisiesto
function esBisiesto(year) {
  return (year % 4 === 0 && year % 100 !== 0) || year % 400 === 0;
}

// 31. Función que encuentra el número de veces que aparece un elemento en un array
function contarOcurrencias(arr, elemento) {
  return arr.filter(item => item === elemento).length;
}

// 32. Función para reemplazar todos los espacios de una cadena por guiones
function reemplazarEspacios(str) {
  return str.replace(/\s/g, '-');
}

// 33. Función que calcula la media geométrica de un array de números
function mediaGeometrica(arr) {
  const producto = arr.reduce((acc, num) => acc * num, 1);
  return Math.pow(producto, 1 / arr.length);
}

// 34. Función para obtener el nombre del mes a partir de un número
function obtenerMes(num) {
  const meses = [
    'Enero', 'Febrero', 'Marzo', 'Abril', 'Mayo', 'Junio',
    'Julio', 'Agosto', 'Septiembre', 'Octubre', 'Noviembre', 'Diciembre'
  ];
  return meses[num - 1] || 'Número inválido';
}

// 35. Función que devuelve los primeros n números de Fibonacci
function fibonacci(n) {
  const secuencia = [0, 1];
  for (let i = 2; i < n; i++) {
    secuencia.push(secuencia[i - 1] + secuencia[i - 2]);
  }
  return secuencia.slice(0, n);
}

// 36. Función para calcular la diferencia absoluta entre dos números
function diferenciaAbsoluta(a, b) {
  return Math.abs(a - b);
}

// 37. Función que verifica si todas las letras de una cadena están en mayúsculas
function estaEnMayusculas(str) {
  return str === str.toUpperCase();
}

// 38. Función para calcular el descuento de un producto
function calcularDescuento(precio, descuento) {
  return precio - (precio * (descuento / 100));
}

// 39. Función que mezcla dos arrays alternando sus elementos
function mezclarArrays(arr1, arr2) {
  const resultado = [];
  const longitud = Math.max(arr1.length, arr2.length);
  for (let i = 0; i < longitud; i++) {
    if (i < arr1.length) resultado.push(arr1[i]);
    if (i < arr2.length) resultado.push(arr2[i]);
  }
  return resultado;
}

// 40. Función para calcular la suma de los dígitos de un número
function sumarDigitos(num) {
  return String(num).split('').reduce((suma, digito) => suma + parseInt(digito), 0);
}
```

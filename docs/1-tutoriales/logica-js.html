Pensamiento Algorítmico

El pensamiento algorítmico es la capacidad de resolver problemas mediante una secuencia lógica de pasos. Es la base de toda programación.
Problema: Sumar dos números
// Algoritmo en pseudocódigo
1. INICIO
2. LEER numero1
3. LEER numero2
4. resultado = numero1 + numero2
5. MOSTRAR resultado
6. FIN

// Implementación en JavaScript
function sumar(a, b) {
    return a + b;
}
Explicación: Todo algoritmo sigue una estructura básica: entrada, proceso, salida.
Problema: Determinar si un número es par
// Pseudocódigo
1. INICIO
2. LEER numero
3. SI numero % 2 == 0 ENTONCES
4.     MOSTRAR "Es par"
5. SINO
6.     MOSTRAR "Es impar"
7. FIN SI
8. FIN

// JavaScript
function esPar(numero) {
    if (numero % 2 === 0) {
        return true;
    } else {
        return false;
    }
}
Explicación: Usamos el operador módulo (%) para obtener el residuo de la división.

Operadores Lógicos

Los operadores lógicos nos permiten combinar o modificar expresiones booleanas (verdaderas o falsas).
AND (&&) - Conjunción
// Ambas condiciones deben ser verdaderas
let edad = 25;
let tieneLicencia = true;

if (edad >= 18 && tieneLicencia) {
    console.log("Puede conducir");
} else {
    console.log("No puede conducir");
}

// Tabla de verdad para AND
// true && true = true
// true && false = false
// false && true = false
// false && false = false
OR (||) - Disyunción
// Al menos una condición debe ser verdadera
let esEstudiante = false;
let esEmpleado = true;

if (esEstudiante || esEmpleado) {
    console.log("Tiene descuento");
}

// Tabla de verdad para OR
// true || true = true
// true || false = true
// false || true = true
// false || false = false
NOT (!) - Negación
// Invierte el valor booleano
let llueve = true;

if (!llueve) {
    console.log("Puedo salir sin paraguas");
} else {
    console.log("Necesito paraguas");
}

// Tabla de verdad para NOT
// !true = false
// !false = true
💡 Consejo de Lógica
Leyes de De Morgan:
1. !(A && B) = !A || !B
2. !(A || B) = !A && !B
Úsalas para simplificar expresiones lógicas complejas.

Estructuras Condicionales

Permiten que el programa tome decisiones y ejecute diferentes bloques de código según condiciones.
if - else if - else
// Sistema de calificaciones
let puntaje = 85;

if (puntaje >= 90) {
    console.log("A - Excelente");
} else if (puntaje >= 80) {
    console.log("B - Bueno");
} else if (puntaje >= 70) {
    console.log("C - Regular");
} else if (puntaje >= 60) {
    console.log("D - Aprobado");
} else {
    console.log("F - Reprobado");
}

// Output: "B - Bueno"
Operador Ternario
// Sintaxis compacta para if-else simples
let edad = 20;
let mensaje = (edad >= 18) ? "Mayor de edad" : "Menor de edad";

console.log(mensaje); // "Mayor de edad"

// También se puede anidar
let temperatura = 25;
let estado = (temperatura > 30) ? "Caluroso" : 
             (temperatura < 15) ? "Frío" : "Agradable";

console.log(estado); // "Agradable"

Bucles (Loops)

Permiten ejecutar un bloque de código múltiples veces. Esenciales para procesar colecciones de datos.
for - Bucle con contador
// Imprimir números del 1 al 5
for (let i = 1; i <= 5; i++) {
    console.log("Número:", i);
}

// Recorrer un array
let frutas = ["Manzana", "Banana", "Naranja"];
for (let i = 0; i < frutas.length; i++) {
    console.log(frutas[i]);
}

// Output:
// Número: 1
// Número: 2
// ...
while - Bucle con condición
// Mientras se cumpla la condición
let contador = 0;

while (contador < 5) {
    console.log("Contador:", contador);
    contador++; // Incrementar para evitar loop infinito
}

// Adivinar número secreto
let numeroSecreto = 7;
let intento = 0;

while (intento !== numeroSecreto) {
    intento = parseInt(prompt("Adivina el número (1-10):"));
    
    if (intento < numeroSecreto) {
        console.log("Más alto");
    } else if (intento > numeroSecreto) {
        console.log("Más bajo");
    }
}

console.log("¡Correcto!");

Patrones de Lógica Comunes

Patrón Acumulador
// Sumar todos los elementos de un array
let numeros = [1, 2, 3, 4, 5];
let suma = 0; // Acumulador

for (let i = 0; i < numeros.length; i++) {
    suma += numeros[i]; // Acumular
}

console.log("Suma total:", suma); // 15

// Multiplicar todos los elementos
let productos = [2, 3, 4];
let productoTotal = 1; // Iniciar en 1 para multiplicación

for (let num of productos) {
    productoTotal *= num;
}

console.log("Producto total:", productoTotal); // 24
Patrón Contador
// Contar elementos que cumplen una condición
let edades = [15, 22, 17, 30, 19, 16];
let mayores = 0; // Contador

for (let edad of edades) {
    if (edad >= 18) {
        mayores++; // Incrementar contador
    }
}

console.log("Personas mayores de edad:", mayores);

// Contar vocales en un string
let texto = "JavaScript es genial";
let vocales = 0;
let textoLower = texto.toLowerCase();

for (let letra of textoLower) {
    if ('aeiouáéíóú'.includes(letra)) {
        vocales++;
    }
}

console.log("Número de vocales:", vocales);
🚀 Patrón Banderas/Flags
Las banderas son variables booleanas que indican si algo ha ocurrido o no. Útiles para validaciones. 
// Verificar si todos los números son positivos
let numeros = [5, 2, 8, -1, 7];
let todosPositivos = true; // Bandera

for (let num of numeros) {
    if (num < 0) {
        todosPositivos = false;
        break; // Salir del bucle temprano
    }
}

console.log("¿Todos positivos?", todosPositivos); // false

📝 Ejercicios Prácticos

Ejercicio 1: Encontrar el máximo
// Encuentra el número mayor en un array
function encontrarMaximo(numeros) {
    // Tu código aquí
    let maximo = numeros[0];
    
    for (let i = 1; i < numeros.length; i++) {
        if (numeros[i] > maximo) {
            maximo = numeros[i];
        }
    }
    
    return maximo;
}

// Prueba
console.log(encontrarMaximo([3, 8, 2, 9, 5])); // 9
Ejercicio 2: Palíndromo
// Determinar si una palabra es palíndromo
function esPalindromo(palabra) {
    // Tu código aquí
    let invertida = "";
    
    for (let i = palabra.length - 1; i >= 0; i--) {
        invertida += palabra[i];
    }
    
    return palabra === invertida;
}

// Pruebas
console.log(esPalindromo("reconocer")); // true
console.log(esPalindromo("javascript")); // false

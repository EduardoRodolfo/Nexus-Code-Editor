## 🚀 **Archivo `integration.js` Corregido y Mejorado:**

```javascript
class EditorTerminalIntegration {
    constructor() {
        this.terminal = null;
        this.editor = null;
        this.packageManager = null;
        this.isTerminalVisible = true;
        
        this.init();
    }

    async init() {
        await this.waitForComponents();
        this.setupIntegration();
        this.setupModalHandlers();
        console.log('✅ Integración editor-terminal inicializada');
    }

    async waitForComponents() {
        return new Promise((resolve) => {
            const checkComponents = () => {
                if (window.terminal && window.codeEditor && window.packageManager) {
                    this.terminal = window.terminal;
                    this.editor = window.codeEditor;
                    this.packageManager = window.packageManager;
                    resolve();
                } else {
                    setTimeout(checkComponents, 100);
                }
            };
            checkComponents();
        });
    }

    setupIntegration() {
        // Toggle de terminal
        document.getElementById('terminal-toggle').addEventListener('click', () => {
            this.toggleTerminal();
        });

        // Ejecutar código
        document.getElementById('run-btn').addEventListener('click', () => {
            this.runFromEditor();
        });

        // Guardar
        document.getElementById('save-btn').addEventListener('click', () => {
            this.saveCurrentFile();
        });

        // Limpiar terminal
        document.getElementById('clear-terminal').addEventListener('click', () => {
            if (this.terminal) this.terminal.clear();
        });

        // Instalar paquetes
        document.getElementById('install-pkg').addEventListener('click', () => {
            this.showPackageModal();
        });

        this.setupEditorTabs();
    }

    setupModalHandlers() {
        const modal = document.getElementById('package-modal');
        const closeBtn = document.querySelector('.close');
        const installBtn = document.getElementById('install-btn');

        closeBtn.addEventListener('click', () => {
            modal.style.display = 'none';
        });

        installBtn.addEventListener('click', () => {
            this.installPackagesFromModal();
        });

        // Cerrar modal al hacer clic fuera
        window.addEventListener('click', (e) => {
            if (e.target === modal) {
                modal.style.display = 'none';
            }
        });

        // Sugerencias de paquetes
        document.querySelectorAll('.suggestion').forEach(suggestion => {
            suggestion.addEventListener('click', () => {
                document.getElementById('package-input').value = suggestion.getAttribute('data-pkg');
            });
        });
    }

    toggleTerminal() {
        const panel = document.getElementById('terminal-panel');
        this.isTerminalVisible = !this.isTerminalVisible;
        
        if (this.isTerminalVisible) {
            panel.classList.remove('collapsed');
            // Reajustar terminal cuando se muestra
            setTimeout(() => {
                if (this.terminal && this.terminal.fitAddon) {
                    this.terminal.fitAddon.fit();
                }
            }, 100);
        } else {
            panel.classList.add('collapsed');
        }
    }

    runFromEditor() {
        if (this.terminal) {
            this.terminal.runCurrentCode();
        }
    }

    saveCurrentFile() {
        if (this.editor) {
            this.editor.saveCurrentFile();
        }
    }

    setupEditorTabs() {
        document.querySelectorAll('.editor-tabs .tab').forEach(tab => {
            tab.addEventListener('click', (e) => {
                const filename = e.target.getAttribute('data-file');
                this.switchEditorFile(filename);
                
                document.querySelectorAll('.editor-tabs .tab').forEach(t => {
                    t.classList.remove('active');
                });
                e.target.classList.add('active');
            });
        });
    }

    switchEditorFile(filename) {
        if (this.editor) {
            this.editor.switchFile(filename);
        }
    }

    showPackageModal() {
        const modal = document.getElementById('package-modal');
        const input = document.getElementById('package-input');
        modal.style.display = 'block';
        input.focus();
        input.value = '';
    }

    async installPackagesFromModal() {
        const input = document.getElementById('package-input');
        const packages = input.value.split(' ').filter(pkg => pkg.trim());
        
        if (packages.length === 0) {
            alert('Por favor ingresa al menos un paquete');
            return;
        }

        const modal = document.getElementById('package-modal');
        modal.style.display = 'none';

        if (this.terminal && this.packageManager) {
            try {
                this.terminal.terminal.write(`\r\n\x1b[36m📦 Instalando: ${packages.join(', ')}\x1b[0m\r\n`);
                const results = await this.packageManager.install(packages);
                
                results.forEach(result => {
                    if (result.success) {
                        this.terminal.terminal.write(`\x1b[32m✅ ${result.name} instalado correctamente\x1b[0m\r\n`);
                    } else {
                        this.terminal.terminal.write(`\x1b[31m❌ Error con ${result.name}: ${result.error}\x1b[0m\r\n`);
                    }
                });
                
                this.terminal.prompt();
            } catch (error) {
                this.terminal.terminal.write(`\x1b[31m❌ Error durante la instalación: ${error.message}\x1b[0m\r\n`);
                this.terminal.prompt();
            }
        }
    }
}
```

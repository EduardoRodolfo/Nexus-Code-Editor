## **🚀 Normalizador de Proyectos Descargados**

### **Código Completo del Normalizador**
```javascript
class ProjectNormalizer {
  constructor() {
    this.originalHTML = '';
    this.normalizedHTML = '';
    this.combinedCSS = '';
    this.combinedJS = '';
  }

  /**
   * Normaliza un proyecto descargado para que funcione en tu editor
   */
  normalizeDownloadedProject(html, css = '', js = '') {
    this.originalHTML = html;
    this.combinedCSS = css;
    this.combinedJS = js;

    // Paso 1: Extraer CSS interno y externo
    this.extractAllCSS();
    
    // Paso 2: Extraer JS interno y externo  
    this.extractAllJS();
    
    // Paso 3: Limpiar HTML (quitar links y scripts externos)
    this.cleanHTML();

    return {
      html: this.normalizedHTML,
      css: this.combinedCSS,
      js: this.combinedJS
    };
  }

  /**
   * Extrae todo el CSS (interno y simula externo)
   */
  extractAllCSS() {
    const html = this.originalHTML;

    // 1. Extraer CSS interno (<style> tags)
    const internalCSS = this.extractInternalCSS(html);
    if (internalCSS) {
      this.combinedCSS += '\n' + internalCSS;
    }

    // 2. Extraer CSS externo (convertir a interno)
    this.extractExternalCSS(html);
  }

  /**
   * Extrae CSS de tags <style>
   */
  extractInternalCSS(html) {
    const styleRegex = /<style[^>]*>([\s\S]*?)<\/style>/gi;
    const matches = [];
    let match;

    while ((match = styleRegex.exec(html)) !== null) {
      matches.push(match[1].trim());
    }

    return matches.join('\n');
  }

  /**
   * Maneja CSS externo - para proyectos descargados lo agregamos como comentario
   */
  extractExternalCSS(html) {
    const linkRegex = /<link[^>]*href=["']([^"']*\.css)["'][^>]*>/gi;
    let match;

    while ((match = linkRegex.exec(html)) !== null) {
      const cssPath = match[1];
      // Agregar como comentario para que el usuario sepa que había CSS externo
      this.combinedCSS += `\n/* CSS externo original: ${cssPath} */\n`;
      this.combinedCSS += `/* Nota: El archivo ${cssPath} necesita ser incorporado manualmente */\n`;
    }
  }

  /**
   * Extrae todo el JavaScript (interno y simula externo)
   */
  extractAllJS() {
    const html = this.originalHTML;

    // 1. Extraer JS interno (<script> sin src)
    const internalJS = this.extractInternalJS(html);
    if (internalJS) {
      this.combinedJS += '\n' + internalJS;
    }

    // 2. Manejar JS externo
    this.extractExternalJS(html);
  }

  /**
   * Extrae JS de tags <script> sin src
   */
  extractInternalJS(html) {
    const scriptRegex = /<script(?!.*src)[^>]*>([\s\S]*?)<\/script>/gi;
    const matches = [];
    let match;

    while ((match = scriptRegex.exec(html)) !== null) {
      matches.push(match[1].trim());
    }

    return matches.join('\n');
  }

  /**
   * Maneja JS externo - para proyectos descargados
   */
  extractExternalJS(html) {
    const scriptRegex = /<script[^>]*src=["']([^"']*\.js)["'][^>]*><\/script>/gi;
    let match;

    while ((match = scriptRegex.exec(html)) !== null) {
      const jsPath = match[1];
      // Agregar como comentario para que el usuario sepa que había JS externo
      this.combinedJS += `\n// JS externo original: ${jsPath}\n`;
      this.combinedJS += `// Nota: El archivo ${jsPath} necesita ser incorporado manualmente\n`;
    }
  }

  /**
   * Limpia el HTML quitando referencias externas
   */
  cleanHTML() {
    let html = this.originalHTML;

    // Quitar links a CSS externos
    html = html.replace(/<link[^>]*href=["'][^"']*\.css["'][^>]*>/gi, '');
    
    // Quitar scripts externos
    html = html.replace(/<script[^>]*src=["'][^"']*\.js["'][^>]*><\/script>/gi, '');
    
    // Quitar tags style (ya extraímos el contenido)
    html = html.replace(/<style[^>]*>[\s\S]*?<\/style>/gi, '');
    
    // Quitar scripts internos (ya extraímos el contenido)
    html = html.replace(/<script(?!.*src)[^>]*>[\s\S]*?<\/script>/gi, '');

    this.normalizedHTML = html.trim();
  }
}
```

function initializeEditor() {
    if (!editor) {
        editor = ace.edit("editor");
        editor.setTheme("ace/theme/dreamweaver");
        editor.session.setMode("ace/mode/html");
        
        // Configuración inicial
        editor.setOptions({
            wrap: true,
            enableBasicAutocompletion: true,
            enableLiveAutocompletion: true,
            enableSnippets: true,
            fontSize: 14,
            showPrintMargin: false,
            useSoftTabs: true,
            tabSize: 2
        });
        
        // Función para actualizar la previsualización
        const updatePreview = () => {
            const preview = document.getElementById('preview');
            const code = editor.getValue();
            preview.srcdoc = code;
        };

        // Actualizar previsualización al cambiar el editor
        editor.session.on('change', () => {
            updatePreview();
        });

        return editor;
    }
    return editor;
}

        }

        extractAllCSS() {
            const html = this.originalHTML;
            const internalCSS = this.extractInternalCSS(html);
            if (internalCSS) {
                this.combinedCSS += '\n' + internalCSS;
            }
            this.extractExternalCSS(html);
        }

        extractInternalCSS(html) {
            const styleRegex = /<style[^>]*>([\s\S]*?)<\/style>/gi;
            const matches = [];
            let match;
            while ((match = styleRegex.exec(html)) !== null) {
                matches.push(match[1].trim());
            }
            return matches.join('\n');
        }

        extractExternalCSS(html) {
            const linkRegex = /<link[^>]*href=["']([^"']*\.css)["'][^>]*>/gi;
            let match;
            while ((match = linkRegex.exec(html)) !== null) {
                const cssPath = match[1];
                this.combinedCSS += `\n/* CSS externo original: ${cssPath} */\n`;
                this.combinedCSS += `/* Nota: El archivo ${cssPath} necesita ser incorporado manualmente */\n`;
            }
        }

        extractAllJS() {
            const html = this.originalHTML;
            const internalJS = this.extractInternalJS(html);
            if (internalJS) {
                this.combinedJS += '\n' + internalJS;
            }
            this.extractExternalJS(html);
        }

        extractInternalJS(html) {
            const scriptRegex = /<script(?!.*src)[^>]*>([\s\S]*?)<\/script>/gi;
            const matches = [];
            let match;
            while ((match = scriptRegex.exec(html)) !== null) {
                matches.push(match[1].trim());
            }
            return matches.join('\n');
        }

        extractExternalJS(html) {
            const scriptRegex = /<script[^>]*src=["']([^"']*\.js)["'][^>]*><\/script>/gi;
            let match;
            while ((match = scriptRegex.exec(html)) !== null) {
                const jsPath = match[1];
                this.combinedJS += `\n// JS externo original: ${jsPath}\n`;
                this.combinedJS += `// Nota: El archivo ${jsPath} necesita ser incorporado manualmente\n`;
            }
        }

        cleanHTML() {
            let html = this.originalHTML;
            html = html.replace(/<link[^>]*href=["'][^"']*\.css["'][^>]*>/gi, '');
            html = html.replace(/<script[^>]*src=["'][^"']*\.js["'][^>]*><\/script>/gi, '');
            html = html.replace(/<style[^>]*>[\s\S]*?<\/style>/gi, '');
            html = html.replace(/<script(?!.*src)[^>]*>[\s\S]*?<\/script>/gi, '');
            this.normalizedHTML = html.trim();
        }
    }

    // Abrir archivo
    async function openFile() {
        if (!isFileSystemAccessSupported()) {
            showNotification('La API de acceso a archivos no está soportada en este navegador.', true);
            return;
        }

        try {
            const [handle] = await window.showOpenFilePicker({
                types: [{
                    description: 'Archivos de texto',
                    accept: {'text/plain': ['.txt', '.js', '.html', '.css', '.json']}
                }]
            });
            
            fileHandle = handle;
            const file = await fileHandle.getFile();
            const contents = await file.text();
            currentFileName = file.name;
            
            // Actualizar modo según extensión
            const extension = currentFileName.split('.').pop();
            const modeMap = {
                'js': 'javascript',
                'html': 'html',
                'css': 'css',
                'json': 'json'
            };
            
            if (modeMap[extension]) {
                editor.session.setMode(`ace/mode/${modeMap[extension]}`);
                document.getElementById('modeSelector').value = modeMap[extension];
            }
            
            editor.setValue(contents);
            showNotification(`Archivo abierto: ${currentFileName}`);
        } catch (error) {
            if (error.name !== 'AbortError') {
                console.error('Error abriendo archivo:', error);
                showNotification(`Error: ${error.message}`, true);
            }
        }
    }

    // Guardar archivo

        if (!isFileSystemAccessSupported()) {
            showNotification('La API de acceso a archivos no está soportada en este navegador.', true);

        try {
            if (!fileHandle) {
                return await saveFileAs();

        }
    }

    // Listar archivos recursivamente
    async function listFiles(directoryHandle, container, path = '') {
        for await (const [name, handle] of directoryHandle.entries()) {
            const fileItem = document.createElement('div');
            fileItem.className = 'file-item';
            fileItem.dataset.path = path ? `${path}/${name}` : name;
            
            const isDirectory = handle.kind === 'directory';
            const icon = isDirectory ? '📁' : getFileIcon(name);
            
            // Crear la estructura del elemento de archivo/carpeta
            fileItem.innerHTML = `
                <span class="folder-toggle">${isDirectory ? '▶' : ''}</span>
                <span class="file-icon">${icon}</span>
                <span class="file-name">${name}</span>
            `;

            if (isDirectory) {
                fileItem.classList.add('folder-item');
                
                // Crear contenedor para hijos
                const folderChildren = document.createElement('div');
                folderChildren.className = 'folder-children';
                fileItem.appendChild(folderChildren);
                
                // Agregar evento para expandir/colapsar
                const toggle = fileItem.querySelector('.folder-toggle');
                fileItem.addEventListener('click', async (e) => {
                    e.stopPropagation();
                    
                    if (!fileItem.classList.contains('expanded')) {
                        // Expandir y cargar contenido
                        fileItem.classList.add('expanded');
                        toggle.textContent = '▼';
                        await listFiles(handle, folderChildren, path ? `${path}/${name}` : name);
                    } else {
                        // Colapsar
                        fileItem.classList.remove('expanded');
                        toggle.textContent = '▶';
                    }
                });
            } else {
                // Es un archivo, agregar evento para abrirlo
                fileItem.addEventListener('click', (e) => {
                    e.stopPropagation();
                    openFileFromProject(handle, name);
                    
                    // Remover active de todos y agregar al actual
                    document.querySelectorAll('.file-item').forEach(item => item.classList.remove('active'));
                    fileItem.classList.add('active');
                });
            }

            container.appendChild(fileItem);
            projectFiles.push({ name, handle, path, kind: handle.kind });
        }
    }

    // Crear pestaña de archivo
    function createFileTab(filename, content) {
        let tabsContainer = document.querySelector('.editor-tabs');
        if (!tabsContainer) {
            tabsContainer = document.createElement('div');
            tabsContainer.className = 'editor-tabs';
            document.querySelector('.editor-container').insertBefore(tabsContainer, document.getElementById('editor'));
        }

        const tab = document.createElement('div');
        tab.className = 'editor-tab';
        tab.innerHTML = `
            <span>${filename}</span>
            <span class="close-tab">×</span>
        `;

        tab.addEventListener('click', (e) => {
            if (e.target.classList.contains('close-tab')) {
                closeFileTab(filename, tab);
                return;
            }
            switchToTab(filename, tab);
        });

        tabsContainer.appendChild(tab);
        switchToTab(filename, tab);
        
        // Actualizar editor con contenido del archivo
        editor.setValue(content);
        
        // Configurar modo según extensión
        const ext = filename.split('.').pop().toLowerCase();
        const modeMap = {
            'js': 'javascript', 'javascript': 'javascript',
            'html': 'html', 'htm': 'html',
            'css': 'css',
            'json': 'json'
        };
        
        if (modeMap[ext]) {
            editor.session.setMode(`ace/mode/${modeMap[ext]}`);
            document.getElementById('modeSelector').value = modeMap[ext];
        }
    }

    // Cambiar a pestaña
    function switchToTab(filename, tab) {
        // Remover clase active de todas las pestañas
        document.querySelectorAll('.editor-tab').forEach(t => t.classList.remove('active'));
        
        // Activar pestaña actual
        tab.classList.add('active');
        currentFileTab = filename;
        
        // Cargar contenido del archivo
        const fileData = openFiles.get(filename);
        if (fileData) {
            editor.setValue(fileData.content);
        }
    }

    // Cerrar pestaña de archivo
    function closeFileTab(filename, tab) {
        const fileData = openFiles.get(filename);
        if (fileData && fileData.modified) {
            if (!confirm('¿Guardar cambios antes de cerrar?')) {
                return;
            }
            saveCurrentFile();
        }
        
        openFiles.delete(filename);
        tab.remove();
        
        // Si no hay pestañas, limpiar editor
        if (openFiles.size === 0) {
            editor.setValue('');
            currentFileTab = null;
        }
    }

                return target[prop] || window[prop];

                // Ejecutar el código

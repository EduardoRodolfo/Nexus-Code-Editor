    constructor() {
        this.initializeElements();
        this.setupEventListeners();
        this.commandHistory = [];
        this.historyIndex = -1;
        this.isInitialized = false;
    }

    initializeElements() {

        this.consoleOutput = document.getElementById('consoleOutput');
        this.runButton = document.getElementById('runButton');
        this.clearConsole = document.getElementById('clearConsole');

            return;
        }
        
        this.isInitialized = true;
        console.log('✅ ConsolaPrincipal inicializada correctamente');
    }

    setupEventListeners() {
        if (!this.isInitialized) return;

        // Botón ejecutar código del editor
        this.runButton.addEventListener('click', () => {
            this.executeEditorCode();
        });

        // Botón limpiar consola
        this.clearConsole.addEventListener('click', () => {
            this.clearConsoleOutput();
        });

        // Atajos de teclado
        document.addEventListener('keydown', (e) => {
            if (e.ctrlKey && e.key === 'Enter') {
                e.preventDefault();
                this.executeEditorCode();
            }
            
            if (e.ctrlKey && e.key === 'l') {
                e.preventDefault();
                this.clearConsoleOutput();
            }
        });

        console.log('✅ Event listeners de ConsolaPrincipal configurados');
    }

    executeEditorCode() {
        if (!window.editor) {
            this.addToConsole('❌ Editor no disponible', 'error');
            return;
        }

        const code = window.editor.getValue().trim();
        const language = document.getElementById('modeSelector')?.value || 'html';
        
        if (language !== "javascript") {
            this.addToConsole("⚠️ La consola solo ejecuta código JavaScript\nCambia el lenguaje a JavaScript para ejecutar código", 'warning');
            return;
        }

        if (!code) {
            this.addToConsole('❌ No hay código para ejecutar', 'error');
            return;
        }

        this.addToConsole('▶️ Ejecutando código del editor...', 'info');
        this.executeJavaScriptCode(code);
    }

    async executeJavaScriptCode(code) {
        try {
            // Crear un contexto aislado más seguro
            const sandbox = this.createSecureSandbox();
            
            // Función para ejecutar en el contexto aislado
            const execute = () => {
                'use strict';
                return new Promise((resolve) => {
                    const AsyncFunction = Object.getPrototypeOf(async function(){}).constructor;
                    const fn = new AsyncFunction('sandbox', `
                        with(sandbox) {
                            try {
                                ${code}
                            } catch(e) {
                                console.error('Error:', e.message);
                            }
                        }
                    `);
                    
                    resolve(fn(sandbox));
                });
            };

            // Ejecutar con timeout controlado
            await Promise.race([
                execute(),
                new Promise((_, reject) => setTimeout(
                    () => reject(new Error("⏰ Timeout excedido (10 segundos)")), 
                    10000
                ))
            ]);
            
        } catch (error) {
            this.addToConsole(`❌ Error de ejecución: ${error.message}`, 'error');
        }
    }

    createSecureSandbox() {
        const sandbox = {
            // Console methods
            console: {
                log: (...args) => {
                    const message = args.map(arg => 
                        typeof arg === 'object' ? JSON.stringify(arg, null, 2) : String(arg)
                    ).join(' ');
                    this.addToConsole(message, 'log');
                },
                error: (...args) => {
                    const message = args.map(arg => 
                        typeof arg === 'object' ? JSON.stringify(arg, null, 2) : String(arg)
                    ).join(' ');
                    this.addToConsole(`❌ ${message}`, 'error');
                },
                warn: (...args) => {
                    const message = args.map(arg => 
                        typeof arg === 'object' ? JSON.stringify(arg, null, 2) : String(arg)
                    ).join(' ');
                    this.addToConsole(`⚠️ ${message}`, 'warning');
                },
                info: (...args) => {
                    const message = args.map(arg => 
                        typeof arg === 'object' ? JSON.stringify(arg, null, 2) : String(arg)
                    ).join(' ');
                    this.addToConsole(`ℹ️ ${message}`, 'info');
                }
            },
            
            // Safe globals
            setTimeout: setTimeout,
            setInterval: setInterval,
            clearTimeout: clearTimeout,
            clearInterval: clearInterval,
            Promise: window.Promise,
            Math: Math,
            Date: Date,
            JSON: JSON,
            Number: Number,
            String: String,
            Boolean: Boolean,
            Array: Array,
            Object: Object,
            
            // Restricted access
            window: undefined,
            document: undefined,
            eval: undefined
        };

        return new Proxy(sandbox, {
            has: () => true,
            get: (target, prop) => {
                if (prop === Symbol.unscopables) return undefined;
                if (prop in target) return target[prop];
                return undefined; // Bloquear acceso a propiedades no permitidas
            }
        });
    }

    addToConsole(text, type = 'normal') {
        if (!this.consoleOutput) return;
        
        const timestamp = new Date().toLocaleTimeString();
        let formattedText = '';

        switch (type) {
            case 'error':
                formattedText = `[${timestamp}] ❌ ${text}\n`;
                break;
            case 'warning':
                formattedText = `[${timestamp}] ⚠️ ${text}\n`;
                break;
            case 'info':
                formattedText = `[${timestamp}] ℹ️ ${text}\n`;
                break;
            case 'success':
                formattedText = `[${timestamp}] ✅ ${text}\n`;
                break;
            case 'log':
                formattedText = `[${timestamp}] ${text}\n`;
                break;
            default:
                formattedText = `[${timestamp}] ${text}\n`;
        }

        this.consoleOutput.value += formattedText;
        this.consoleOutput.scrollTop = this.consoleOutput.scrollHeight;
    }

    clearConsoleOutput() {
        if (this.consoleOutput) {
            this.consoleOutput.value = '';
            this.addToConsole('🧹 Consola limpiada', 'info');
        }
    }
}

class ConsolaAlterna {
    constructor() {
        this.consoleId = 'alternate';
        this.initializeElements();
        this.setupEventListeners();

        this.isInitialized = false;

    initializeElements() {
        // Elementos de la consola alterna
        this.consoleInput = document.getElementById('alternateConsoleInput');
        this.consoleOutput = document.getElementById('alternateConsoleOutput');
        this.runButton = document.getElementById('runAlternateConsole');
        this.clearInputButton = document.getElementById('clearAlternateInput');
        this.clearOutputButton = document.getElementById('clearAlternateOutput');
        
        if (!this.consoleInput || !this.consoleOutput || !this.runButton) {
            console.warn('Elementos de consola alterna no encontrados');
            return;
        }
        
        this.isInitialized = true;
        console.log('✅ ConsolaAlterna inicializada correctamente');
    }

    setupEventListeners() {
        if (!this.isInitialized) return;

        // Botón ejecutar código
        this.runButton.addEventListener('click', () => {
            this.executeInputCode();
        });

        // Botón limpiar input
        this.clearInputButton.addEventListener('click', () => {
            this.clearInput();
        });

        // Botón limpiar output
        this.clearOutputButton.addEventListener('click', () => {
            this.clearOutput();
        });

        // Atajos de teclado específicos para consola alterna
        this.consoleInput.addEventListener('keydown', (e) => {
            if (e.ctrlKey && e.key === 'Enter') {
                e.preventDefault();
                this.executeInputCode();
            }
            
            if (e.ctrlKey && e.key === 'l' && e.shiftKey) {
                e.preventDefault();
                this.clearOutput();
            }
            
            // Navegación por historial (↑/↓)
            if (e.key === 'ArrowUp') {
                e.preventDefault();
                this.navigateHistory(-1);
            }
            
            if (e.key === 'ArrowDown') {
                e.preventDefault();
                this.navigateHistory(1);
            }
        });

        // Auto-expandir textarea del input
        this.consoleInput.addEventListener('input', () => {
            this.autoResizeTextarea();
        });

        console.log('✅ Event listeners de ConsolaAlterna configurados');
    }

    executeInputCode() {
        const code = this.consoleInput.value.trim();
        
        if (!code) {
            this.addToOutput('❌ No hay código para ejecutar', 'error');
            return;
        }

        // Agregar al historial
        this.addToHistory(code);
        
        this.addToOutput('> ' + code, 'input');
        this.executeJavaScriptCode(code);

    async executeJavaScriptCode(code) {
        try {
            // Crear contexto de ejecución
            const sandbox = this.createSecureSandbox();
            
            // Función para ejecutar en el contexto aislado
            const execute = () => {
                'use strict';
                return new Promise((resolve) => {
                    const AsyncFunction = Object.getPrototypeOf(async function(){}).constructor;
                    const fn = new AsyncFunction('sandbox', `

                            try {
                                return (function() {
                                    ${code}
                                })();
                            } catch(e) {
                                console.error('Error:', e.message);
                                return undefined;
                            }
                        }

                    ).join(' ');

                },
                error: (...args) => {
                    const message = args.map(arg => 
                        typeof arg === 'object' ? JSON.stringify(arg, null, 2) : String(arg)
                    ).join(' ');

                        }

                        this.addToOutput(`📊 ${String(data)}`, 'log');
                    }
                },

                    this.clearOutput();

            Math: Math,
            JSON: JSON,
            Date: Date,
            Number: Number,

            Boolean: Boolean,
            Array: Array,

        this.commandHistory.unshift(command);
        if (this.commandHistory.length > 50) {

        this.historyIndex = -1;

        if (this.commandHistory.length === 0) return;
        
        this.historyIndex += direction;
        
        if (this.historyIndex < 0) {
            this.historyIndex = -1;
            this.consoleInput.value = '';
        } else if (this.historyIndex >= this.commandHistory.length) {
            this.historyIndex = this.commandHistory.length - 1;
        } else {
            this.consoleInput.value = this.commandHistory[this.historyIndex];
        }
    }

    formatAsTable(data) {
        if (Array.isArray(data)) {
            if (data.length === 0) return '(array vacío)';
            
            // Para arrays simples
            if (data.every(item => typeof item !== 'object')) {
                return data.map((item, index) => `${index}: ${item}`).join('\n');
            }
            
            // Para arrays de objetos
            const headers = Object.keys(data[0] || {});
            if (headers.length === 0) return '(sin propiedades)';
            
            let table = headers.join('\t');
            table += '\n' + headers.map(() => '---').join('\t');
            
            data.forEach(item => {
                const row = headers.map(header => 
                    item[header] !== undefined ? String(item[header]) : ''
                );
                table += '\n' + row.join('\t');
            });
            
            return table;
        } else {
            // Para objetos
            return Object.entries(data)
                .map(([key, value]) => `${key}: ${JSON.stringify(value)}`)
                .join('\n');
        }
    }

    // Método para inyectar variables globales seguras
    injectGlobal(name, value) {
        if (this.executionContext) {
            this.executionContext[name] = value;
        }
    }

    // Método para ejecutar código de forma programática
    async executeCode(code) {
        this.consoleInput.value = code;
        await this.executeInputCode();
    }
}

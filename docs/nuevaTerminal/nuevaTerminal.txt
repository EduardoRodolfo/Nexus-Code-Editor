## 3. **Terminal JavaScript (scripts/terminal.js)**

```javascript
class IntegratedTerminal {
    constructor() {
        this.terminal = null;
        this.fitAddon = null;
        this.isInitialized = false;
        this.currentDir = '~/proyecto';
        this.commandHistory = [];
        this.installedPackages = [];
        
        this.init();
    }

    async init() {
        try {
            // Inicializar terminal
            this.terminal = new Terminal({
                theme: {
                    background: '#1e1e1e',
                    foreground: '#ffffff',
                    cursor: '#ffffff',
                    selection: '#3a3d41',
                    black: '#000000',
                    red: '#cd3131',
                    green: '#0dbc79',
                    yellow: '#e5e510',
                    blue: '#2472c8',
                    magenta: '#bc3fbc',
                    cyan: '#11a8cd',
                    white: '#e5e5e5'
                },
                fontSize: 12,
                fontFamily: 'Consolas, "Courier New", monospace',
                cursorBlink: true,
                scrollback: 1000
            });

            this.fitAddon = new FitAddon.FitAddon();
            this.terminal.loadAddon(this.fitAddon);

            // Abrir terminal en el contenedor
            this.terminal.open(document.getElementById('terminal'));
            this.fitAddon.fit();

            // Configurar eventos
            this.setupEventListeners();
            this.showWelcomeMessage();
            
            this.isInitialized = true;
            
        } catch (error) {
            console.error('Error inicializando terminal:', error);
        }
    }

    setupEventListeners() {
        // Entrada del usuario
        this.terminal.onData((data) => this.handleInput(data));

        // Botones de la interfaz
        document.getElementById('clear-terminal').addEventListener('click', () => this.clear());
        document.getElementById('run-code').addEventListener('click', () => this.runCurrentCode());
        document.getElementById('install-pkg').addEventListener('click', () => this.showPackageModal());
        
        // Tabs de terminal
        document.querySelectorAll('.term-tab').forEach(tab => {
            tab.addEventListener('click', (e) => this.switchTab(e.target));
        });

        // Redimensionamiento
        window.addEventListener('resize', () => this.fitAddon.fit());
    }

    handleInput(data) {
        const code = data.charCodeAt(0);
        
        if (code === 13) { // Enter
            this.processCommand(this.currentLine);
            this.currentLine = '';
        } else if (code === 127) { // Backspace
            if (this.currentLine.length > 0) {
                this.terminal.write('\b \b');
                this.currentLine = this.currentLine.slice(0, -1);
            }
        } else if (code >= 32 && code <= 126) { // Caracteres imprimibles
            this.terminal.write(data);
            this.currentLine += data;
        }
    }

    processCommand(command) {
        this.terminal.write('\r\n');
        this.commandHistory.push(command);

        const args = command.trim().split(' ').filter(arg => arg);
        const cmd = args[0]?.toLowerCase();

        switch(cmd) {
            case 'clear':
                this.clear();
                break;
            case 'run':
                this.runCurrentCode();
                break;
            case 'install':
                this.installPackage(args.slice(1));
                break;
            case 'ls':
                this.listFiles();
                break;
            case 'help':
                this.showHelp();
                break;
            default:
                if (cmd) this.executeSystemCommand(command);
        }

        if (cmd !== 'clear') {
            this.prompt();
        }
    }

    executeSystemCommand(command) {
        // Simular ejecución de comandos del sistema
        this.terminal.write(`\x1b[33mEjecutando: ${command}\x1b[0m\r\n`);
        
        // Simular procesamiento
        setTimeout(() => {
            this.terminal.write(`\x1b[32mComando completado exitosamente\x1b[0m\r\n`);
        }, 500);
    }

    async runCurrentCode() {
        const code = document.getElementById('code-editor').value;
        
        this.terminal.write('\r\n\x1b[36m🚀 Ejecutando código...\x1b[0m\r\n');
        
        // Simular ejecución del código
        this.simulateProgress('Compilando', 30);
        
        setTimeout(() => {
            this.simulateProgress('Ejecutando', 40);
            
            setTimeout(() => {
                this.terminal.write('\r\n\x1b[32m✅ Código ejecutado correctamente\x1b[0m\r\n');
                
                // Mostrar "salida" simulada
                if (code.includes('console.log')) {
                    this.terminal.write('\x1b[37mSalida: ¡Hola desde el editor!\x1b[0m\r\n');
                }
                
                if (code.includes('app.listen')) {
                    this.terminal.write('\x1b[37m🌐 Servidor iniciado en http://localhost:3000\x1b[0m\r\n');
                }
                
                this.prompt();
            }, 800);
        }, 600);
    }

    simulateProgress(message, steps) {
        let current = 0;
        const interval = setInterval(() => {
            current++;
            const percent = Math.min(100, Math.round((current / steps) * 100));
            const bars = '█'.repeat(Math.round(percent / 2));
            const spaces = ' '.repeat(50 - Math.round(percent / 2));
            
            this.terminal.write(`\r\x1b[36m${message}: [${bars}${spaces}] ${percent}%\x1b[0m`);
            
            if (current >= steps) {
                clearInterval(interval);
                this.terminal.write('\r\n');
            }
        }, 20);
    }

    installPackage(packages) {
        if (packages.length === 0) {
            this.terminal.write('\x1b[31m❌ Especifica paquetes para instalar\x1b[0m\r\n');
            return;
        }

        this.terminal.write(`\r\n\x1b[36m📦 Instalando: ${packages.join(', ')}\x1b[0m\r\n`);

        packages.forEach((pkg, index) => {
            setTimeout(() => {
                this.simulatePackageInstallation(pkg);
                
                if (index === packages.length - 1) {
                    setTimeout(() => {
                        this.updatePackageStatus();
                        this.terminal.write('\r\n\x1b[32m✅ Instalación completada\x1b[0m\r\n');
                        this.prompt();
                    }, 500);
                }
            }, index * 1000);
        });
    }

    simulatePackageInstallation(packageName) {
        this.terminal.write(`\x1b[34m📥 Descargando ${packageName}...\x1b[0m\r\n`);
        
        this.simulateProgress(`Instalando ${packageName}`, 25);
        
        setTimeout(() => {
            // Agregar a paquetes instalados
            if (!this.installedPackages.includes(packageName)) {
                this.installedPackages.push(packageName);
            }
            
            this.terminal.write(`\x1b[32m✅ ${packageName} instalado\x1b[0m\r\n`);
        }, 600);
    }

    showPackageModal() {
        const modal = document.getElementById('package-modal');
        const input = document.getElementById('package-input');
        
        modal.style.display = 'block';
        input.focus();
        
        // Configurar sugerencias
        document.querySelectorAll('.suggestion').forEach(suggestion => {
            suggestion.addEventListener('click', () => {
                input.value = suggestion.getAttribute('data-pkg');
            });
        });
        
        // Configurar instalación
        document.getElementById('install-btn').onclick = () => {
            const packages = input.value.split(' ').filter(pkg => pkg.trim());
            if (packages.length > 0) {
                this.installPackage(packages);
                modal.style.display = 'none';
                input.value = '';
            }
        };
        
        // Cerrar modal
        document.querySelector('.close').onclick = () => {
            modal.style.display = 'none';
        };
    }

    listFiles() {
        const files = [
            'src/',
            'public/',
            'package.json',
            'README.md',
            'index.js',
            'styles.css'
        ];
        
        this.terminal.write('\r\n');
        files.forEach(file => {
            if (file.endsWith('/')) {
                this.terminal.write(`\x1b[34m${file}\x1b[0m\r\n`);
            } else {
                this.terminal.write(`\x1b[32m${file}\x1b[0m\r\n`);
            }
        });
    }

    showHelp() {
        const help = `
\x1b[36mComandos disponibles:\x1b[0m
\x1b[32mhelp\x1b[0m      - Muestra esta ayuda
\x1b[32mclear\x1b[0m     - Limpia la terminal
\x1b[32mrun\x1b[0m       - Ejecuta el código actual
\x1b[32minstall\x1b[0m    - Instala paquetes npm
\x1b[32mls\x1b[0m        - Lista archivos

\x1b[36mBotones de interfaz:\x1b[0m
▶️ Ejecutar   - Ejecuta el código
📦 Instalar  - Instala paquetes
🧹 Limpiar   - Limpia la terminal
`;
        this.terminal.write(help);
    }

    showWelcomeMessage() {
        const welcome = `
\x1b[36m🚀 Terminal Integrada - CodeStudio\x1b[0m
Escribe 'help' para ver los comandos disponibles
Usa los botones para acciones rápidas

`;
        this.terminal.write(welcome);
        this.prompt();
    }

    prompt() {
        this.terminal.write(`\r\n\x1b[32muser@codestudio\x1b[0m:\x1b[34m${this.currentDir}\x1b[0m$ `);
        this.currentLine = '';
    }

    clear() {
        this.terminal.clear();
        this.prompt();
    }

    switchTab(tabElement) {
        // Remover active de todos los tabs
        document.querySelectorAll('.term-tab').forEach(tab => {
            tab.classList.remove('active');
        });
        
        // Activar tab clickeado
        tabElement.classList.add('active');
        
        const terminalType = tabElement.getAttribute('data-terminal');
        
        // Aquí puedes cambiar la vista según el tab
        switch(terminalType) {
            case 'logs':
                this.showLogsView();
                break;
            case 'packages':
                this.showPackagesView();
                break;
            default:
                this.showMainTerminal();
        }
    }

    showLogsView() {
        this.terminal.write('\x1b[33m📊 Cambiando a vista de logs...\x1b[0m\r\n');
    }

    showPackagesView() {
        this.terminal.write('\x1b[33m📦 Cambiando a gestión de paquetes...\x1b[0m\r\n');
        if (this.installedPackages.length > 0) {
            this.terminal.write('Paquetes instalados:\r\n');
            this.installedPackages.forEach(pkg => {
                this.terminal.write(`  📦 ${pkg}\r\n`);
            });
        } else {
            this.terminal.write('No hay paquetes instalados\r\n');
        }
    }

    showMainTerminal() {
        this.terminal.write('\x1b[33m💻 Volviendo a terminal principal...\x1b[0m\r\n');
    }

    updatePackageStatus() {
        const count = this.installedPackages.length;
        document.getElementById('package-count').textContent = `Paquetes: ${count}`;
    }

    // Método para integración con el editor
    executeInTerminal(command) {
        this.terminal.write(`\r\n\x1b[33m$$ {command}\x1b[0m\r\n`);
        this.processCommand(command);
    }
}

// Inicializar cuando el DOM esté listo
document.addEventListener('DOMContentLoaded', () => {
    window.terminal = new IntegratedTerminal();
});
```

## 4. **Integración (scripts/integration.js)**

```javascript
class EditorTerminalIntegration {
    constructor() {
        this.terminal = null;
        this.isTerminalVisible = true;
        
        this.init();
    }

    init() {
        // Esperar a que la terminal esté lista
        setTimeout(() => {
            this.terminal = window.terminal;
            this.setupIntegration();
        }, 1000);
    }

    setupIntegration() {
        // Toggle de terminal
        document.getElementById('terminal-toggle').addEventListener('click', () => {
            this.toggleTerminal();
        });

        // Ejecutar código desde el editor
        document.getElementById('run-btn').addEventListener('click', () => {
            this.runFromEditor();
        });

        // Guardar y ejecutar
        document.getElementById('save-btn').addEventListener('click', () => {
            this.saveAndRun();
        });

        // Tabs del editor
        this.setupEditorTabs();
    }

    toggleTerminal() {
        const panel = document.getElementById('terminal-panel');
        this.isTerminalVisible = !this.isTerminalVisible;
        
        if (this.isTerminalVisible) {
            panel.classList.remove('collapsed');
        } else {
            panel.classList.add('collapsed');
        }
    }

    runFromEditor() {
        if (this.terminal) {
            this.terminal.runCurrentCode();
        }
    }

    saveAndRun() {
        // Simular guardado
        const code = document.getElementById('code-editor').value;
        localStorage.setItem('current-code', code);
        
        // Mostrar mensaje
        if (this.terminal) {
            this.terminal.terminal.write('\r\n\x1b[32m💾 Código guardado\x1b[0m\r\n');
            this.terminal.runCurrentCode();
        }
    }

    setupEditorTabs() {
        document.querySelectorAll('.editor-tabs .tab').forEach(tab => {
            tab.addEventListener('click', (e) => {
                const file = e.target.getAttribute('data-file');
                this.switchFile(file);
                
                // Actualizar tabs activos
                document.querySelectorAll('.editor-tabs .tab').forEach(t => {
                    t.classList.remove('active');
                });
                e.target.classList.add('active');
            });
        });
    }

    switchFile(filename) {
        const editor = document.getElementById('code-editor');
        
        // Simular cambio de archivo
        const fileContents = {
            'index.js': `// ${filename}
const express = require('express');
const app = express();

app.get('/', (req, res) => {
    res.send('Hola desde ${filename}');
});

app.listen(3000, () => {
    console.log('Servidor ejecutándose');
});`,
            'package.json': `{
  "name": "mi-proyecto",
  "version": "1.0.0",
  "dependencies": {
    "express": "^4.18.0"
  },
  "scripts": {
    "start": "node index.js"
  }
}`,
            'styles.css': `/* ${filename} */
body {
    font-family: Arial, sans-serif;
    margin: 0;
    padding: 20px;
    background: #f5f5f5;
}

.container {
    max-width: 1200px;
    margin: 0 auto;
}`
        };

        editor.value = fileContents[filename] || `// ${filename}\n// Contenido del archivo`;
    }
}

// Inicializar integración
document.addEventListener('DOMContentLoaded', () => {
    window.editorIntegration = new EditorTerminalIntegration();
});
```

```javascript
    async install(packageNames) {

        for (const packageName of packageNames) {
            try {
                const result = await this.installSinglePackage(packageName);
                results.push(result);

    async installSinglePackage(packageName) {
        const pkgInfo = this.availablePackages.get(packageName);
        
        if (!pkgInfo) {
            throw new Error(`Paquete no encontrado: ${packageName}`);
        }

        // Verificar si ya está instalado
        if (this.installedPackages.has(packageName)) {
            const installed = this.installedPackages.get(packageName);
            return {
                name: packageName,
                success: true,
                skipped: true,
                message: `Ya instalado (v${installed.version})`
            };
        }

        // Simular instalación
        await this.simulateInstallation(packageName, pkgInfo);

        // Instalar dependencias si las tiene
        if (pkgInfo.dependencies) {
            for (const dep of pkgInfo.dependencies) {
                if (!this.installedPackages.has(dep)) {
                    await this.installSinglePackage(dep);
                }
            }
        }

        // Registrar como instalado
        this.installedPackages.set(packageName, {
            ...pkgInfo,
            installedAt: new Date().toISOString(),
            installedVersion: pkgInfo.version
        });

        return {
            name: packageName,
            success: true,
            version: pkgInfo.version,
            size: pkgInfo.size
        };
    }

    async simulateInstallation(packageName, pkgInfo) {
        // Simular diferentes fases de instalación
        const phases = [
            { name: 'Resolviendo dependencias', duration: 300 },
            { name: 'Descargando paquete', duration: 500 },
            { name: 'Extrayendo archivos', duration: 400 },
            { name: 'Construyendo paquete', duration: 600 },
            { name: 'Verificando integridad', duration: 200 }
        ];

        for (const phase of phases) {
            if (window.terminal) {
                window.terminal.terminal.write(`\x1b[34m📦 ${phase.name} ${packageName}...\x1b[0m\r\n`);
            }
            await this.delay(phase.duration);
        }
    }

    delay(ms) {
        return new Promise(resolve => setTimeout(resolve, ms));
    }

    uninstall(packageNames) {
        const results = [];
        
        for (const packageName of packageNames) {
            try {
                const success = this.installedPackages.delete(packageName);
                results.push({
                    name: packageName,
                    success: success,
                    message: success ? 'Desinstalado correctamente' : 'No estaba instalado'
                });
            } catch (error) {
                results.push({
                    name: packageName,
                    success: false,
                    error: error.message
                });
            }
        }
        
        this.saveInstalledPackages();
        this.updatePackageJson();
        
        return results;
    }

    search(query) {
        const results = [];
        const queryLower = query.toLowerCase();
        
        for (const [name, pkg] of this.availablePackages) {
            if (name.toLowerCase().includes(queryLower) || 
                pkg.description.toLowerCase().includes(queryLower)) {
                results.push({
                    ...pkg,
                    installed: this.installedPackages.has(name)
                });
            }
        }
        
        return results.sort((a, b) => {
            // Ordenar por relevancia (coincidencia exacta primero)
            if (a.name === query) return -1;
            if (b.name === query) return 1;
            return a.name.localeCompare(b.name);
        });
    }

    getInstalledPackages() {
        return Array.from(this.installedPackages.values());
    }

    getOutdatedPackages() {
        const outdated = [];
        
        for (const [name, installed] of this.installedPackages) {
            const available = this.availablePackages.get(name);
            if (available && available.version !== installed.version) {
                outdated.push({
                    name,
                    current: installed.version,
                    latest: available.version,
                    installed: installed.installedAt
                });
            }
        }
        
        return outdated;
    }

    updatePackage(packageNames) {
        const results = [];
        
        for (const packageName of packageNames) {
            const available = this.availablePackages.get(packageName);
            const installed = this.installedPackages.get(packageName);
            
            if (!available || !installed) {
                results.push({
                    name: packageName,
                    success: false,
                    error: 'Paquete no encontrado'
                });
                continue;
            }
            
            if (available.version === installed.version) {
                results.push({
                    name: packageName,
                    success: true,
                    skipped: true,
                    message: 'Ya está en la versión más reciente'
                });
                continue;
            }
            
            // Actualizar versión
            this.installedPackages.set(packageName, {
                ...installed,
                version: available.version,
                installedVersion: available.version,
                updatedAt: new Date().toISOString()
            });
            
            results.push({
                name: packageName,
                success: true,
                from: installed.version,
                to: available.version
            });
        }
        
        this.saveInstalledPackages();
        this.updatePackageJson();
        
        return results;
    }

    updatePackageJson() {
        // Actualizar package.json con las dependencias instaladas
        const dependencies = {};
        
        for (const [name, pkg] of this.installedPackages) {
            dependencies[name] = `^${pkg.version}`;
        }
        
        // Si existe el editor, actualizar package.json
        if (window.codeEditor && window.codeEditor.files['package.json']) {
            try {
                const currentContent = window.codeEditor.files['package.json'];
                const packageJson = JSON.parse(currentContent);
                
                packageJson.dependencies = dependencies;
                
                // Actualizar en el editor
                window.codeEditor.files['package.json'] = JSON.stringify(packageJson, null, 2);
                
                // Si estamos viendo package.json, actualizar la vista
                if (window.codeEditor.currentFile === 'package.json') {
                    window.codeEditor.loadFile('package.json');
                }
                
            } catch (error) {
                console.error('Error actualizando package.json:', error);
            }
        }
    }

    saveInstalledPackages() {
        try {
            const packages = this.getInstalledPackages();
            localStorage.setItem('codestudio-installed-packages', JSON.stringify(packages));
        } catch (error) {
            console.error('Error guardando paquetes instalados:', error);
        }
    }

    getPackageInfo(packageName) {
        return this.availablePackages.get(packageName) || null;
    }

    getStats() {
        const total = this.availablePackages.size;
        const installed = this.installedPackages.size;
        const outdated = this.getOutdatedPackages().length;
        
        return {
            total,
            installed,
            outdated,
            available: total - installed
        };
    }

    // Método para integración con la terminal
    executePackageCommand(command, args) {
        switch(command) {
            case 'list':
                return this.getInstalledPackages();
            case 'search':
                return this.search(args[0] || '');
            case 'outdated':
                return this.getOutdatedPackages();
            case 'stats':
                return this.getStats();
            default:
                throw new Error(`Comando no reconocido: ${command}`);
        }
    }
}

    window.packageManager = new PackageManager();
```

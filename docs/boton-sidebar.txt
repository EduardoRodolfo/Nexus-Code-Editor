```javascript
// =====================================
// Funciones para mostrar/ocultar sidebar
// =====================================

function toggleProjectSidebar() {
    const sidebar = document.getElementById('project-sidebar');
    
    if (currentProjectHandle) {
        // Si hay proyecto abierto, alternar visibilidad
        if (sidebar.style.display === 'none' || sidebar.style.display === '') {
            showProjectSidebar();
        } else {
            hideProjectSidebar();
        }
    } else {
        // Si no hay proyecto, preguntar si abrir uno
        if (confirm('No hay proyecto abierto. ¿Deseas abrir un proyecto?')) {
            openProject();
        }
    }
}

function showProjectSidebar() {
    if (!currentProjectHandle) {
        showNotification('No hay proyecto abierto', true);
        return;
    }
    
    document.getElementById('project-sidebar').style.display = 'block';
    // Asegurarnos de que el contenido principal se ajuste
    const mainContent = document.querySelector('.container');
    if (mainContent) {
        mainContent.style.marginLeft = '300px';
    }
    showNotification('Explorador de proyecto abierto');
}

function hideProjectSidebar() {
    document.getElementById('project-sidebar').style.display = 'none';
    const mainContent = document.querySelector('.container');
    if (mainContent) {
        mainContent.style.marginLeft = '0';
    }
}

// Modificar la función closeProjectSidebar existente
function closeProjectSidebar() {
    hideProjectSidebar();
    // No cerramos el proyecto, solo ocultamos el sidebar
    showNotification('Explorador de proyecto cerrado');
}
```

```javascript
// En newProject y openProject, después de loadProjectStructure():
async function newProject() {
    if (!isFileSystemAccessSupported()) {
        showNotification('La API de acceso a archivos no está soportada en este navegador.', true);
        return;
    }

    try {
        currentProjectHandle = await window.showDirectoryPicker();
        await loadProjectStructure();
        // Mostrar sidebar automáticamente al crear proyecto
        showProjectSidebar();
        showNotification('Nuevo proyecto creado');
        
        await createDefaultProjectStructure();
    } catch (error) {
        if (error.name !== 'AbortError') {
            console.error('Error creando proyecto:', error);
            showNotification(`Error: ${error.message}`, true);
        }
    }
}

async function openProject() {
    if (!isFileSystemAccessSupported()) {
        showNotification('La API de acceso a archivos no está soportada en este navegador.', true);
        return;
    }

    try {
        currentProjectHandle = await window.showDirectoryPicker();
        await loadProjectStructure();
        // Mostrar sidebar automáticamente al abrir proyecto
        showProjectSidebar();
        showNotification('Proyecto abierto correctamente');
    } catch (error) {
        if (error.name !== 'AbortError') {
            console.error('Error abriendo proyecto:', error);
            showNotification(`Error: ${error.message}`, true);
        }
    }
}
```

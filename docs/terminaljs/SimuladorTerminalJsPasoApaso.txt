handleAdminCommands(cmd, args) {
    switch(cmd) {
        case 'sudo':
            if (args.length > 0) {
                this.terminal.writeln('\x1B[1;33m[SUDO] Ejecutando como administrador...\x1B[0m');
                // Simular ejecución del comando con sudo
                const sudoCmd = args[0];
                const sudoArgs = args.slice(1);
                
                if (sudoCmd === 'apt' || sudoCmd === 'install' || sudoCmd === 'update' || sudoCmd === 'upgrade') {
                    this.systemState.isRoot = true;
                    this.handleAdminCommands(sudoCmd, sudoArgs);
                    this.systemState.isRoot = false;
                } else {
                    this.terminal.writeln(`\x1B[32mComando '${sudoCmd}' ejecutado con privilegios de administrador\x1B[0m`);
                }
            } else {
                this.terminal.writeln('\x1B[31mUso: sudo <comando> [argumentos]\x1B[0m');
            }
            return true;
            
        case 'apt':
            if (args.length === 0) {
                this.terminal.writeln('\x1B[1;36mGestor de paquetes APT - Comandos disponibles:\x1B[0m');
                this.terminal.writeln('  update    - Actualizar lista de paquetes');
                this.terminal.writeln('  upgrade   - Actualizar paquetes instalados');
                this.terminal.writeln('  install   - Instalar nuevos paquetes');
                this.terminal.writeln('  list      - Listar paquetes disponibles');
                return true;
            }
            return this.handleAptCommands(args[0], args.slice(1));
            
        case 'update':
            this.simulateSystemUpdate();
            return true;
            
        case 'upgrade':
            this.simulateSystemUpgrade();
            return true;
            
        case 'install':
            if (args.length > 0) {
                this.simulatePackageInstall(args);
            } else {
                this.terminal.writeln('\x1B[31mUso: install <paquete1> [paquete2...]\x1B[0m');
            }
            return true;
            
        case 'whoami':
            this.terminal.writeln(this.systemState.user);
            return true;
            
        case 'su':
            this.terminal.writeln('\x1B[33mCambiando a usuario root...\x1B[0m');
            this.systemState.user = 'root';
            this.systemState.isRoot = true;
            this.terminal.writeln('\x1B[32mAhora eres root\x1B[0m');
            return true;
            
        case 'exit':
            if (this.systemState.isRoot) {
                this.terminal.writeln('\x1B[33mSaliendo de sesión root...\x1B[0m');
                this.systemState.user = 'usuario-terminal';
                this.systemState.isRoot = false;
            }
            return true;
    }
    return false;
}

// ⭐⭐ NUEVO MÉTODO: Comandos específicos de APT
handleAptCommands(subcmd, args) {
    switch(subcmd) {
        case 'update':
            this.simulateAptUpdate();
            return true;
            
        case 'upgrade':
            this.simulateAptUpgrade();
            return true;
            
        case 'install':
            if (args.length > 0) {
                this.simulateAptInstall(args);
            } else {
                this.terminal.writeln('\x1B[31mUso: apt install <paquete1> [paquete2...]\x1B[0m');
            }
            return true;
            
        case 'list':
            this.terminal.writeln('\x1B[1;36mPaquetes instalados:\x1B[0m');
            this.systemState.packages.forEach(pkg => {
                this.terminal.writeln(`  ${pkg} - versión 1.0.0`);
            });
            return true;
            
        default:
            this.terminal.writeln(`\x1B[31mSubcomando APT no reconocido: ${subcmd}\x1B[0m`);
            return true;
    }
}

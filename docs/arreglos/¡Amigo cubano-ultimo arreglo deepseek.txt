### 2. index.js (corregido y completo)
```javascript
// trigger Editor extensions
ace.require("ace/ext/language_tools");

// Variables globales
let currentFileName = null;
let fileHandle = null;
let currentDirectoryHandle = null;

// Verificar si las APIs están disponibles
const isFileSystemAccessSupported = () => {
    return 'showOpenFilePicker' in window && 'showSaveFilePicker' in window && 'showDirectoryPicker' in window;
};

// Función para mostrar notificaciones
function showNotification(message, isError = false) {
    const notification = document.createElement('div');
    notification.className = isError ? 'notification error' : 'notification success';
    notification.textContent = message;
    document.body.appendChild(notification);
    
    setTimeout(() => {
        notification.remove();
    }, 3000);
}

document.addEventListener('DOMContentLoaded', () => {
    // Inicializar Ace Editor
    let editor = ace.edit("editor");
    editor.setTheme("ace/theme/monokai");
    editor.session.setMode("ace/mode/html");
    
    // Configuración inicial
    editor.setOptions({
        enableBasicAutocompletion: true,
        enableLiveAutocompletion: true,
        enableSnippets: true,
        fontSize: 14,
        showPrintMargin: false,
        useSoftTabs: true,
        tabSize: 2
    });
    
    // Función para actualizar la previsualización
    const updatePreview = () => {
        const preview = document.getElementById('preview');
        const code = editor.getValue();
        preview.srcdoc = code;
    };

    // Actualizar previsualización al cambiar el editor
    editor.session.on('change', () => {
        updatePreview();
    });

    // ===== CONFIGURACIÓN DEL EDITOR =====
    document.getElementById('themeSelector').addEventListener('change', (e) => {
        editor.setTheme(`ace/theme/${e.target.value}`);
    });

    document.getElementById('modeSelector').addEventListener('change', (e) => {
        editor.session.setMode(`ace/mode/${e.target.value}`);
    });

    document.getElementById('fontSize').addEventListener('change', (e) => {
        editor.setFontSize(parseInt(e.target.value));
    });

    document.getElementById('lineNumbers').addEventListener('change', (e) => {
        editor.setOption('showLineNumbers', e.target.checked);
    });

    document.getElementById('wrapMode').addEventListener('change', (e) => {
        editor.session.setUseWrapMode(e.target.checked);
    });

    // ===== EVENTOS DEL MENÚ =====
    document.getElementById('abrir-documento').addEventListener('click', openFile);
    document.getElementById('guardar').addEventListener('click', saveFile);
    document.getElementById('guardar-como').addEventListener('click', saveFileAs);
    document.getElementById('nuevo-documento').addEventListener('click', newDocument);
    document.getElementById('nueva-carpeta').addEventListener('click', createNewFolder);
    document.getElementById('limpiar-editor').addEventListener('click', clearEditor);

    // =====================================
    // Funciones para manejo de archivos
    // =====================================

    // Abrir archivo
    async function openFile() {
        if (!isFileSystemAccessSupported()) {
            showNotification('La API de acceso a archivos no está soportada en este navegador.', true);
            return;
        }

        try {
            const [handle] = await window.showOpenFilePicker({
                types: [{
                    description: 'Archivos de texto',
                    accept: {'text/plain': ['.txt', '.js', '.html', '.css', '.json']}
                }]
            });
            
            fileHandle = handle;
            const file = await fileHandle.getFile();
            const contents = await file.text();
            currentFileName = file.name;
            
            // Actualizar modo según extensión
            const extension = currentFileName.split('.').pop();
            const modeMap = {
                'js': 'javascript',
                'html': 'html',
                'css': 'css',
                'json': 'json'
            };
            
            if (modeMap[extension]) {
                editor.session.setMode(`ace/mode/${modeMap[extension]}`);
                document.getElementById('modeSelector').value = modeMap[extension];
            }
            
            editor.setValue(contents);
            showNotification(`Archivo abierto: ${currentFileName}`);
        } catch (error) {
            if (error.name !== 'AbortError') {
                console.error('Error abriendo archivo:', error);
                showNotification(`Error: ${error.message}`, true);
            }
        }
    }

    // Guardar archivo
    async function saveFile() {
        if (!isFileSystemAccessSupported()) {
            showNotification('La API de acceso a archivos no está soportada en este navegador.', true);
            return false;
        }

        try {
            if (!fileHandle) {
                return await saveFileAs();
            }
            
            const writable = await fileHandle.createWritable();
            await writable.write(editor.getValue());
            await writable.close();
            
            showNotification(`Archivo guardado: ${currentFileName || 'sin título'}`);
            return true;
        } catch (error) {
            console.error('Error guardando archivo:', error);
            showNotification(`Error al guardar: ${error.message}`, true);
            return false;
        }
    }

    // Guardar como
    async function saveFileAs() {
        if (!isFileSystemAccessSupported()) {
            showNotification('La API de acceso a archivos no está soportada en este navegador.', true);
            return false;
        }

        try {
            fileHandle = await window.showSaveFilePicker({
                types: [{
                    description: 'Archivos de texto',
                    accept: {'text/plain': ['.txt', '.js', '.html', '.css', '.json']}
                }],
                suggestedName: currentFileName || 'sin_titulo.txt'
            });
            
            const saved = await saveFile();
            if (saved) {
                currentFileName = fileHandle.name;
            }
            return saved;
        } catch (error) {
            if (error.name !== 'AbortError') {
                console.error('Error guardando archivo:', error);
                showNotification(`Error: ${error.message}`, true);
            }
            return false;
        }
    }

    // Nuevo documento
    function newDocument() {
        if (editor.getValue() && !confirm('¿Crear nuevo documento? Se perderán los cambios no guardados.')) return;
        editor.setValue('');
        fileHandle = null;
        currentFileName = 'nuevo_documento.txt';
        editor.session.setMode("ace/mode/text");
        document.getElementById('modeSelector').value = 'html';
        showNotification('Nuevo documento creado');
    }

    // Limpiar editor
    function clearEditor() {
        if (!editor.getValue()) return;
        if (confirm('¿Limpiar editor? Se perderá todo el contenido.')) {
            editor.setValue('');
            showNotification('Editor limpiado');
        }
    }

    // Crear nueva carpeta
    async function createNewFolder() {
        if (!isFileSystemAccessSupported()) {
            showNotification('La API de acceso a archivos no está soportada en este navegador.', true);
            return;
        }

        try {
            if (!currentDirectoryHandle) {
                currentDirectoryHandle = await window.showDirectoryPicker();
            }
            
            const folderName = prompt('Nombre de la nueva carpeta:', 'Nueva Carpeta');
            if (!folderName) return;
            
            await currentDirectoryHandle.getDirectoryHandle(folderName, { create: true });
            showNotification(`Carpeta creada: ${folderName}`);
        } catch (error) {
            if (error.name !== 'AbortError') {
                console.error('Error creando carpeta:', error);
                showNotification(`Error: ${error.message}`, true);
            }
        }
    }

    // Contenido inicial
    editor.setValue(`<!DOCTYPE html>
<html>
<head>
  <title>Ejemplo</title>
  <script>
    console.log("Hola desde el editor");
  </script>
</head>
<body>
  <h1>¡Consola JavaScript Funcionando!</h1>
</body>
</html>`);
    updatePreview();
    
    // ===== CONFIGURACIÓN DE LA CONSOLA =====
    const consoleOutput = document.getElementById('consoleOutput');
    const consoleInput = document.getElementById('consoleInput');
    
    // Historial de comandos
    let commandHistory = [];
    let historyIndex = -1;

    // Limpiar consola
    document.getElementById('clearConsole').addEventListener('click', () => {
        consoleOutput.value = '';
    });

    // Ejecutar código del editor
    document.getElementById('runButton').addEventListener('click', () => {
        const language = document.getElementById('modeSelector').value;
        if (language !== "javascript") {
            addToConsole("La consola solo ejecuta código JavaScript\nCambia el lenguaje a JavaScript para ejecutar código", "error");
            return;
        }
        executeJavaScriptCode(editor.getValue());
    });

    // Configurar input de la consola
    consoleInput.addEventListener('keydown', function(e) {
        if (e.key === 'Enter') {
            const command = consoleInput.value.trim();
            if (command) {
                commandHistory.push(command);
                historyIndex = -1;
                
                addToConsole(`> ${command}`, "input");
                
                try {
                    const result = new Function(command)();
                    if (result !== undefined) {
                        addToConsole(String(result));
                    }
                } catch (error) {
                    addToConsole(`Error: ${error.message}`, "error");
                }
                
                consoleInput.value = '';
            }
        }
        
        // Navegar por el historial
        if (e.key === 'ArrowUp') {
            e.preventDefault();
            if (commandHistory.length > 0 && historyIndex < commandHistory.length - 1) {
                historyIndex++;
                consoleInput.value = commandHistory[commandHistory.length - 1 - historyIndex];
            }
        } else if (e.key === 'ArrowDown') {
            e.preventDefault();
            if (historyIndex > 0) {
                historyIndex--;
                consoleInput.value = commandHistory[commandHistory.length - 1 - historyIndex];
            } else if (historyIndex === 0) {
                historyIndex = -1;
                consoleInput.value = '';
            }
        }
    });

    // Función para añadir texto a la consola
    function addToConsole(text, type = "normal") {
        const timestamp = new Date().toLocaleTimeString();
        const prefix = type === "error" ? `[${timestamp}] ! ERROR: ` :
                     type === "input" ? `[${timestamp}] > ` : 
                     `[${timestamp}] `;
        
        consoleOutput.value += prefix + text + '\n';
        consoleOutput.scrollTop = consoleOutput.scrollHeight;
    }

    // Función para ejecutar código JavaScript (CORREGIDA)
    async function executeJavaScriptCode(code) {
        // Limpiar la consola
        consoleOutput.value = '';
        
        // Guardar los métodos originales de console
        const originalConsole = {
            log: console.log,
            error: console.error,
            warn: console.warn,
            info: console.info,
            debug: console.debug
        };

        // Función para redirigir los logs
        const redirectConsole = (method) => (...args) => {
            originalConsole[method](...args);
            const message = args.map(arg => 
                typeof arg === 'object' ? JSON.stringify(arg, null, 2) : String(arg)
            ).join(' ');
            addToConsole(message, method === 'error' ? 'error' : 'normal');
        };

        // Redirigir todos los métodos
        console.log = redirectConsole('log');
        console.error = redirectConsole('error');
        console.warn = redirectConsole('warn');
        console.info = redirectConsole('info');
        console.debug = redirectConsole('debug');

        try {
            // Crear un contexto de ejecución asíncrono con timeout
            const timeoutPromise = new Promise((_, reject) => 
                setTimeout(() => reject(new Error("Timeout excedido (5 segundos)")), 5000)
            );

            // Crear función asíncrona para ejecutar el código
            const AsyncFunction = Object.getPrototypeOf(async function(){}).constructor;
            const asyncFunction = new AsyncFunction(code);
            
            // Ejecutar con timeout
            await Promise.race([asyncFunction(), timeoutPromise]);
        } catch (error) {
            console.error(`[Execution Error] ${error.message}`);
        } finally {
            // Restaurar la consola después de un pequeño retraso
            setTimeout(() => {
                console.log = originalConsole.log;
                console.error = originalConsole.error;
                console.warn = originalConsole.warn;
                console.info = originalConsole.info;
                console.debug = originalConsole.debug;
            }, 300);
        }
    }
});
```

### 4. server.js (opcional, solo si lo necesitas)
```javascript
const express = require('express');
const cors = require('cors');
const bodyParser = require('body-parser');
const { exec } = require('child_process');

const app = express();
app.use(cors());
app.use(bodyParser.json());

app.post('/run', (req, res) => {
    const { language, code } = req.body;
    let command;

    if (language === "python") {
        command = `python -c "${code}"`;
    } else if (language === "php") {
        command = `php -r "${code}"`;
    } else if (language === "sql") {
        command = `sqlite3 mydatabase.db "${code}"`;
    }

    if (command) {
        exec(command, (error, stdout, stderr) => {
            res.json({ output: error ? stderr : stdout });
        });
    } else {
        res.json({ output: "Lenguaje no soportado en este servidor." });
    }
});

app.listen(3000, () => console.log("Servidor ejecutando en puerto 3000"));
```

Ejemplo de Fetch API

Demostración de cómo usar fetch() para obtener datos de una API

Código JavaScript

async function obtenerDatos() {
  try {
    // Usamos fetch directamente con su contexto original
    const respuesta = await fetch('https://jsonplaceholder.typicode.com/todos/1');
    const datos = await respuesta.json();
    mostrarResultado(datos);
  } catch (error) {
    mostrarError("Error al obtener datos: " + error);
  }
}

// Alternativa segura con contexto bind
const safeFetch = window.fetch.bind(window);

async function obtenerDatosSeguro() {
  try {
    const respuesta = await safeFetch('https://jsonplaceholder.typicode.com/todos/1');
    const datos = await respuesta.json();
    mostrarResultado(datos);
  } catch (error) {
    mostrarError("Error al obtener datos (safe): " + error);
  }
}

🧠 ¿Qué son las funciones en JavaScript?

Las funciones son bloques de código reutilizables que realizan una tarea específica. Son fundamentales en JavaScript porque:

✅ Permiten organizar el código en tareas más pequeñas y manejables
✅ Evitan la repetición de código (principio DRY - Don't Repeat Yourself)
✅ Facilitan la depuración y mantenimiento del código
✅ Son ciudadanos de primera clase (pueden asignarse, pasarse y retornarse)

En JavaScript, las funciones pueden definirse de varias formas y tienen características únicas que las hacen extremadamente poderosas.

Fundamentos de Funciones

Las funciones son el corazón de JavaScript. Comencemos con los conceptos básicos:

Declaración de funciones

// Sintaxis básica
function saludar(nombre) {
    return `¡Hola, ${nombre}!`;
}

console.log(saludar("María")); // ¡Hola, María!

// Características:
// - Se pueden llamar antes de declararse (hoisting)
// - Tienen su propio contexto (this)

Expresiones de funciones

// Asignar una función a una variable
const sumar = function(a, b) {
    return a + b;
};

console.log(sumar(5, 3)); // 8

// Características:
// - No tienen hoisting
// - Pueden ser anónimas (sin nombre)
// - Útiles como callbacks

Funciones flecha (Arrow Functions)

// Sintaxis concisa
const cuadrado = x => x * x;
console.log(cuadrado(4)); // 16

// Con múltiples parámetros
const multiplicar = (a, b) => a * b;

// Con múltiples líneas
const factorial = n => {
    let result = 1;
    for (let i = 2; i <= n; i++) {
        result *= i;
    }
    return result;
};

console.log(factorial(5)); // 120

// Características:
// - Sintaxis más corta
// - No tienen su propio 'this'
// - No tienen objeto 'arguments'

Parámetros y argumentos

// Parámetros por defecto
function crearUsuario(nombre, rol = "usuario") {
    return { nombre, rol };
}

console.log(crearUsuario("Ana")); 
// {nombre: "Ana", rol: "usuario"}

// Parámetros rest (...)
function sumarNumeros(...numeros) {
    return numeros.reduce((sum, num) => sum + num, 0);
}

console.log(sumarNumeros(1, 2, 3, 4)); // 10

// Desestructuración de objetos
function mostrarDatos({nombre, edad}) {
    return `${nombre} tiene ${edad} años`;
}

const persona = {nombre: "Carlos", edad: 30};
console.log(mostrarDatos(persona)); 
// Carlos tiene 30 años

Temas Avanzados

Consejo Profesional
Dominar estos conceptos avanzados te convertirá en un desarrollador JavaScript más competente y te permitirá escribir código más eficiente y mantenible.

Closures

// Un closure es una función que recuerda su entorno léxico
function crearContador() {
    let contador = 0;
    
    return function() {
        contador++;
        return contador;
    };
}

const incrementar = crearContador();
console.log(incrementar()); // 1
console.log(incrementar()); // 2
console.log(incrementar()); // 3

// Características:
// - Mantienen acceso a variables de su ámbito padre
// - Útiles para crear datos privados
// - Base de la programación funcional

IIFE (Immediately Invoked Function Expression)

// Función que se ejecuta inmediatamente
(function() {
    console.log("Esta función se ejecuta inmediatamente");
})();

// Con parámetros
(function(mensaje) {
    console.log(mensaje);
})("¡Hola desde IIFE!");

// Uso moderno (reemplazado por bloques {})
{
    let temporal = "Variable temporal";
    console.log(temporal);
}

// Características:
// - Creaban un ámbito privado antes de ES6
// - Evitaban contaminación del ámbito global
// - Menos necesarias con let/const

Funciones Callback

// Una función que se pasa como argumento a otra
function procesarDatos(datos, callback) {
    // Simulamos procesamiento asíncrono
    setTimeout(() => {
        const resultado = datos.map(item => item * 2);
        callback(resultado);
    }, 1000);
}

procesarDatos([1, 2, 3], resultado => {
    console.log("Resultado procesado:", resultado);
    // Después de 1 segundo: [2, 4, 6]
});

// Uso con array methods
const numeros = [1, 2, 3, 4, 5];
const pares = numeros.filter(num => num % 2 === 0);
console.log(pares); // [2, 4]

// Características:
// - Fundamentales para operaciones asíncronas
// - Base de métodos de arrays (map, filter, etc.)
// - Pueden llevar al "callback hell" si se anidan mucho

Recursión

// Función que se llama a sí misma
function factorial(n) {
    if (n <= 1) return 1;
    return n * factorial(n - 1);
}

console.log(factorial(5)); // 120

// Recursión con memorización (mejora rendimiento)
function fibonacci(n, memo = {}) {
    if (n in memo) return memo[n];
    if (n <= 2) return 1;
    
    memo[n] = fibonacci(n - 1, memo) + fibonacci(n - 2, memo);
    return memo[n];
}

console.log(fibonacci(10)); // 55

// Características:
// - Útil para problemas que pueden dividirse en subproblemas
// - Requiere un caso base para evitar bucles infinitos
// - Puede ser menos eficiente que las soluciones iterativas

Promesas y Async/Await

// Promesa: representa un valor futuro
function cargarDatos() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            const exito = Math.random() > 0.3;
            exito ? resolve("Datos cargados") : reject("Error de carga");
        }, 1000);
    });
}

// Uso con .then() y .catch()
cargarDatos()
    .then(datos => console.log(datos))
    .catch(error => console.error(error));

// Async/Await (sintaxis más limpia)
async function procesar() {
    try {
        const datos = await cargarDatos();
        console.log("Procesando:", datos);
    } catch (error) {
        console.error("Error:", error);
    }
}

procesar();

// Características:
// - Manejan operaciones asíncronas de forma elegante
// - Async/await hace que el código asíncrono parezca síncrono
// - Evitan el "callback hell"

Ejercicio 1: Convertidor de Temperatura

function celsiusToFahrenheit(celsius) {
    return (celsius * 9/5) + 32;

Ejercicio 2: Generador de Contraseñas

// Crea una función que genere contraseñas aleatorias
function generarContraseña(longitud = 12) {
    const caracteres = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789!@#$%^&*()";
    let contraseña = "";
    
    for (let i = 0; i < longitud; i++) {
        const randomIndex = Math.floor(Math.random() * caracteres.length);
        contraseña += caracteres[randomIndex];
    }
    
    return contraseña;
}

console.log("Contraseña:", generarContraseña()); // Ej: "xT8$kL2!pQ9z"

Ejercicio 3: Validador de Palíndromos

// Crea una función que verifique si una palabra es palíndromo
function esPalindromo(palabra) {
    const limpia = palabra.toLowerCase().replace(/[^a-z0-9]/g, '');
    return limpia === limpia.split('').reverse().join('');
}

console.log("reconocer:", esPalindromo("reconocer")); // true
console.log("JavaScript:", esPalindromo("JavaScript")); // false
console.log("Anita lava la tina:", esPalindromo("Anita lava la tina")); // true

Ejercicio 4: Calculadora de Propinas

// Crea una función que calcule propinas con diferentes niveles de servicio
function calcularPropina(total, servicio = "bueno") {
    const niveles = {
        "excelente": 0.20,
        "bueno": 0.15,
        "regular": 0.10,
        "malo": 0.05
    };
    
    const porcentaje = niveles[servicio.toLowerCase()] || 0.15;
    const propina = total * porcentaje;
    const totalConPropina = total + propina;
    
    return {
        propina: propina.toFixed(2),
        total: totalConPropina.toFixed(2),
        porcentaje: `${(porcentaje * 100).toFixed(0)}%`
    };
}

console.log(calcularPropina(100, "excelente"));
// {propina: "20.00", total: "120.00", porcentaje: "20%"}

Ejercicio 5: Temporizador con Closure

// Crea un temporizador con closure que mantenga el estado
function crearTemporizador() {
    let inicio = null;
    let intervalo = null;
    let tiempoTranscurrido = 0;
    
    function iniciar() {
        if (intervalo) return;
        inicio = Date.now() - tiempoTranscurrido;
        intervalo = setInterval(() => {
            tiempoTranscurrido = Date.now() - inicio;
            console.log(`Tiempo: ${(tiempoTranscurrido/1000).toFixed(1)}s`);
        }, 100);
    }
    
    function pausar() {
        if (!intervalo) return;
        clearInterval(intervalo);
        intervalo = null;
    }
    
    function resetear() {
        pausar();
        tiempoTranscurrido = 0;
        console.log("Temporizador reiniciado");
    }
    
    return { iniciar, pausar, resetear };
}

Calculadora de Factorial

Concepto y aplicaciones

El factorial de un número entero no negativo n (denotado n!) es el producto de todos los enteros positivos menores o iguales a n. En matemáticas, los factoriales son fundamentales en combinatoria, probabilidad y estadística. En la naturaleza, aparecen en cálculos de permutaciones genéticas y en la teoría de probabilidades aplicada a fenómenos naturales.

Descripción: Crear una función que calcule el factorial de un número.

Selecciona una solución:

function calcularFactorial(n) { if (n === 0) return 1; let resultado = 1; for (let i = 1; i <= n; i++) { resultado *= i; } return resultado; }
function calcularFactorial(n) { let resultado = 0; // Error: inicializado a 0 for (let i = 1; i <= n; i++) { resultado *= i; } return resultado; }
function calcularFactorial(n) { if (n < 0) return -1; // Error: no maneja caso 0 let resultado = 1; for (let i = n; i > 0; i++) { // Error: i++ en lugar de i-- resultado *= i; } return resultado; }
✅ Correcto! Esta solución maneja correctamente el caso base (0! = 1) y utiliza un bucle for para multiplicar secuencialmente los números.
❌ Incorrecto! Esta solución tiene errores. La primera opción inicializa el resultado a 0 (debería ser 1). La segunda no maneja el caso n=0 y tiene un bucle infinito (i++ en lugar de i--).
function calcularFactorial(n) { if (n === 0) return 1; let resultado = 1; for (let i = 1; i <= n; i++) { resultado *= i; } return resultado; } console.log("5! =", calcularFactorial(5)); console.log("0! =", calcularFactorial(0)); console.log("3! =", calcularFactorial(3));

Validar Palíndromos

Concepto y aplicaciones

Un palíndromo es una palabra, frase o secuencia que se lee igual en ambas direcciones. En biología, las secuencias palindrómicas de ADN son importantes porque son sitios de reconocimiento para enzimas de restricción. Estas secuencias son fundamentales en ingeniería genética y biotecnología.

Descripción: Crear una función que determine si una palabra es palíndromo.

Selecciona una solución:

function esPalindromo(palabra) { const limpia = palabra.toLowerCase().replace(/[^a-z]/g, ''); return limpia === limpia.split('').reverse().join(''); }
function esPalindromo(palabra) { // Error: no convierte a minúsculas ni limpia return palabra === palabra.reverse(); // Error: reverse no existe en strings }
function esPalindromo(palabra) { // Error: solo compara el primer y último carácter return palabra[0] === palabra[palabra.length-1]; }
✅ Correcto! Esta solución limpia la cadena (elimina espacios y caracteres no alfabéticos) y convierte a minúsculas antes de comparar con su reverso.
❌ Incorrecto! La primera opción intenta usar reverse() directamente en un string (no es posible). La segunda solo compara el primer y último carácter, ignorando el resto.
function esPalindromo(palabra) { const limpia = palabra.toLowerCase().replace(/[^a-z]/g, ''); return limpia === limpia.split('').reverse().join(''); } console.log("'reconocer' es palíndromo:", esPalindromo("reconocer")); console.log("'Anita lava la tina' es palíndromo:", esPalindromo("Anita lava la tina")); console.log("'javascript' es palíndromo:", esPalindromo("javascript"));

Secuencia Fibonacci

Concepto y aplicaciones

La secuencia Fibonacci es una serie donde cada número es la suma de los dos anteriores (0, 1, 1, 2, 3, 5, 8...). En la naturaleza, aparece en la disposición de hojas en plantas, en la estructura de piñas, en la concha del nautilus, en la familia de conejos y en la espiral de galaxias, demostrando patrones matemáticos fundamentales en el universo.

Descripción: Crear una función que retorne un array con los primeros n números de Fibonacci.

Selecciona una solución:

function fibonacci(n) { if (n <= 0) return []; if (n === 1) return [0]; const secuencia = [0, 1]; for (let i = 2; i < n; i++) { secuencia.push(secuencia[i-1] + secuencia[i-2]); } return secuencia; }
function fibonacci(n) { // Error: no maneja n=0 ni n=1 const secuencia = [0, 1]; for (let i = 2; i <= n; i++) { // Error: debe ser i < n secuencia.push(secuencia[i-1] + secuencia[i-2]); } return secuencia; }
function fibonacci(n) { // Error: devuelve números incorrectos const secuencia = [1, 1]; // Error: debe empezar con 0,1 for (let i = 2; i < n; i++) { secuencia.push(secuencia[i-1] + secuencia[i-2]); } return secuencia; }
✅ Correcto! Esta solución maneja correctamente los casos base (n=0 y n=1) y genera la secuencia exacta usando un bucle for.
❌ Incorrecto! La primera opción produce un array más largo de lo necesario (usa <= en lugar de <). La segunda comienza con [1,1] en lugar de [0,1], alterando toda la secuencia.
function fibonacci(n) { if (n <= 0) return []; if (n === 1) return [0]; const secuencia = [0, 1]; for (let i = 2; i < n; i++) { secuencia.push(secuencia[i-1] + secuencia[i-2]); } return secuencia; } console.log("Fibonacci(6):", fibonacci(6)); console.log("Fibonacci(1):", fibonacci(1)); console.log("Fibonacci(10):", fibonacci(10));

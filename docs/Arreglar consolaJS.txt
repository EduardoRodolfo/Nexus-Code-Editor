consoleInput.addEventListener('keydown', function(e) {
    if (e.key === 'Enter') {
        const command = consoleInput.value.trim();
        if (command) {
            commandHistory.push(command);
            historyIndex = -1;
            
            addToConsole(`> ${command}`, "input");
            
            try {
                const result = new Function(command)();
                if (result !== undefined) {
                    addToConsole(String(result));
                }
            } catch (error) {
                addToConsole(`Error: ${error.message}`, "error");
            }
            
            consoleInput.value = '';
        }
    }
    
    // Navegar por el historial
    if (e.key === 'ArrowUp') {
        e.preventDefault();
        if (commandHistory.length > 0 && historyIndex < commandHistory.length - 1) {
            historyIndex++;
            consoleInput.value = commandHistory[commandHistory.length - 1 - historyIndex];
        }
    } else if (e.key === 'ArrowDown') {
        e.preventDefault();
        if (historyIndex > 0) {
            historyIndex--;
            consoleInput.value = commandHistory[commandHistory.length - 1 - historyIndex];
        } else if (historyIndex === 0) {
            historyIndex = -1;
            consoleInput.value = '';
        }
    }
});

        e.preventDefault();
        navigateHistory(1);

// Función para ejecutar código de la consola interactiva
async function executeConsoleCode() {
    const code = consoleInput.value.trim();
    if (!code) {
        addToConsole("No hay código para ejecutar", "warning");
        return;
    }

    // Agregar al historial si es diferente al último comando
    if (commandHistory.length === 0 || commandHistory[commandHistory.length - 1] !== code) {
        commandHistory.push(code);
        if (commandHistory.length > 50) { // Limitar historial a 50 elementos
            commandHistory.shift();
        }
    }
    historyIndex = -1;

    addToConsole("> Ejecutando código de consola interactiva...", "input");
    addToConsole(code, "code");

    try {
        await executeJavaScriptCode(code, 'console');
    } catch (error) {
        addToConsole(`Error ejecutando código: ${error.message}`, "error");
    }
}

// Función para navegar por el historial
function navigateHistory(direction) {
    if (commandHistory.length === 0) return;

    if (direction === -1) { // Flecha arriba
        if (historyIndex < commandHistory.length - 1) {
            historyIndex++;
            consoleInput.value = commandHistory[commandHistory.length - 1 - historyIndex];
        }
    } else { // Flecha abajo
        if (historyIndex > 0) {
            historyIndex--;
            consoleInput.value = commandHistory[commandHistory.length - 1 - historyIndex];
        } else if (historyIndex === 0) {
            historyIndex = -1;
            consoleInput.value = '';
        }
    }
}

// Modificar la función addToConsole para soportar más tipos
function addToConsole(text, type = "normal") {
    const timestamp = new Date().toLocaleTimeString();
    let prefix;
    
    switch (type) {
        case "error":
            prefix = `[${timestamp}] ❌ ERROR: `;
            break;
        case "warning":
            prefix = `[${timestamp}] ⚠️ ADVERTENCIA: `;
            break;
        case "info":
            prefix = `[${timestamp}] ℹ️ INFO: `;
            break;
        case "input":
            prefix = `[${timestamp}] > `;
            break;
        case "code":
            prefix = `[${timestamp}] 📝 CÓDIGO:\n`;
            text = text.split('\n').map(line => '  ' + line).join('\n'); // Indentar código
            break;
        case "success":
            prefix = `[${timestamp}] ✅ ÉXITO: `;
            break;
        default:
            prefix = `[${timestamp}] `;
    }
    
    consoleOutput.value += prefix + text + '\n';
    consoleOutput.scrollTop = consoleOutput.scrollHeight;
}
